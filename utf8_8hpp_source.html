24 template <std::size_t N>
25 [[nodiscard]] constexpr std::optional<code_point_range> find_range(
26  const std::array<code_point_range, N>& table,
27  code_point cp) noexcept
28 {
29  auto first = table.begin();
30  [[maybe_unused]] auto last = table.end();
31 
32  auto len = N;
33  while (len > 0) {
34  const auto step = len / 2;
35  auto halfway = first + step;
36 
37  if (cp < halfway->first) {
38  last = halfway;
39  len = step;
40  } else if (cp <= halfway->last) {
41  return *halfway;
42  } else {
43  first = ++halfway;
44  len -= step + 1;
45  }
46  }
47 
48  return {};
49 }

73 [[nodiscard]] inline constexpr std::size_t code_point_size(std::string_view str) noexcept
74 {
75  if (str.empty()) {
76  return 0;
77  }
78 
79  // The MSB determines if it's ASCII or UTF-8
80  const auto first_byte = static_cast<std::uint8_t>(str[0]);
81  if ((first_byte & 0b1000'0000) == 0) {
82  // ASCII
83  return 1;
84  }
85 
86  // Can't think of a way to do this without branches
87  if (first_byte < 0b1101'1111) {
88  return 2;
89  } else if (first_byte < 0b1110'1111) {
90  return 3;
91  }
92 
93  return 4;
94 }

104 [[nodiscard]] inline constexpr std::optional<code_point> code_point_to_number(
105  std::string_view str) noexcept
106 {
107  const auto bytes_to_read = code_point_size(str);
108  if (bytes_to_read == 0) {
109  return {};
110  }
111 
112  if (bytes_to_read == 1) {
113  // ASCII
114  return str[0];
115  }
116 
117  if (str.size() < bytes_to_read) {
118  return {};
119  }
120 
121  constexpr auto subsequent_byte_mask = code_point{0b0011'1111};
122  const auto first_byte = static_cast<std::uint8_t>(str[0]);
123 
124  auto result = first_byte & (0b0001'1111 >> (bytes_to_read - 2));
125  for (auto i = 1u; i < bytes_to_read; ++i) {
126  const auto subsequent_byte = static_cast<code_point>(str[i]);
127 
128  // Move the previous reads up to make space for the subsequent byte's data
129  result <<= 6;
130  result |= subsequent_byte & subsequent_byte_mask;
131  }
132 
133  return result;
134 }

315 {

432  {
433  if (str_.empty()) {

436 
437  // Start by consuming the previous line
438  if (line_break_byte_ != 0) {
439  str_.remove_prefix(line_break_byte_);

445  if (!is_whitespace(cp)) {

458  // break markers at each whitespace code point
459  auto column = std::size_t{0};

462  auto line_break_byte = std::size_t{0};

468  // code point

471  } else {

473  }
474  return;
475  }
476 
477  bytes += code_point_size(cp);
478  if (is_whitespace(cp)) {
479  line_break_column = column;
480  line_break_byte = bytes;
481  }
482  }
483 
484  // We haven't hit the max column, so consume the whole string
485  line_break_byte_ = str_.size();
486  }
487
