18 template <std::size_t N>
19 [[nodiscard]] constexpr std::optional<code_point::range> find_range(
20  const std::array<code_point::range, N>& table,
21  code_point::type cp) noexcept
22 {
23  auto first = table.begin();
24 
25  auto len = N;
26  while (len > 0) {
27  const auto step = len / 2;
28  auto halfway = first + step;
29 
30  if (cp < halfway->first()) {
31  len = step;
32  } else if (cp <= halfway->last()) {
33  return *halfway;
34  } else {
35  first = ++halfway;
36  len -= step + 1;
37  }
38  }
39 
40  return {};
41 }

483  constexpr void consume() noexcept
484  {
485  if (str_.empty()) {
486  return;
487  }
488 
489  // Start by consuming the previous line
490  if (line_break_byte_ != 0) {
491  str_.remove_prefix(line_break_byte_);
492  line_break_byte_ = 0;
493 
494  // Consuming any leading whitespace
495  auto bytes = std::size_t{0};
496  for (auto cp : code_point::iterator::range(str_)) {
497  if (!is_whitespace(cp)) {
498  break;
499  }
500  bytes += code_point::size(cp);
501  }
502  str_.remove_prefix(bytes);
503 
504  if (str_.empty()) {
505  return;
506  }
507  }
508 
509  // Iterate over the code points until you reach or exceed the column limit, setting the
510  // break markers at each whitespace code point
511  auto column = std::size_t{0};
512  auto bytes = std::size_t{0};
513  auto line_break_column = std::size_t{0};
514  auto line_break_byte = std::size_t{0};
515  for (auto it = code_point::iterator{str_}; it != code_point::iterator{}; ++it) {
516  // Have we exceeded the terminal width?
517  column += terminal_width(*it);
518  if (column > max_columns_) {
519  if (line_break_column == 0) {
520  // Oh dear, the line has no whitespace in it to break on, so force break on the
521  // last code point
522  line_break_byte_ = bytes;
523  } else {
524  line_break_byte_ = line_break_byte;
525  }
526  return;
527  }
528 
529  bytes += code_point::size(*it);
530 
531  // If this code point is malformed, then just skip it
532  const auto this_cp = code_point::decode(*it);
533  if (!this_cp) {
534  continue;
535  }

537  rotate_trailing_window();

539 
540  // If there is a following code point, use it to update the next_break_class
541  auto next_break_class = line_break_class::any;

544  ++next_it;

546  const auto next_cp = code_point::decode(*next_it);
