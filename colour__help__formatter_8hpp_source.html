13 template <std::size_t Indent>
14 class colour_line_formatter
15 {
16  static_assert(Indent > 0, "Indent must be greater than zero");
17 
18  constexpr static auto reset_colour = std::string_view{"\033[0m"};
19  constexpr static auto red_colour = std::string_view{"\033[31m"};
20  constexpr static auto green_colour = std::string_view{"\033[32m"};
21 
22 public:
31  template <std::size_t DescStart, std::size_t Depth, typename HelpData>
32  void format(std::ostream& stream, std::size_t columns)
33  {
34  if constexpr (!HelpData::label::empty()) {
35  constexpr auto indent = indent_size<Depth>();
36  stream << red_colour << utility::create_sequence_cts_t<indent, ' '>::get()
37  << HelpData::label::get();
38 
39  if constexpr (!HelpData::description::empty()) {
40  static_assert(HelpData::description::get().find('\t') == std::string_view::npos,
41  "Help descriptions cannot contain tabs");
42 
43  constexpr auto gap =
44  DescStart - indent - utility::utf8::terminal_width(HelpData::label::get());
45 
46  // Spacing between the end of the args label and start of description
47  stream << green_colour << utility::create_sequence_cts_t<gap, ' '>::get();
48 
49  // Print the description, breaking if a word will exceed the terminal width
50  for (auto it = utility::utf8::line_iterator{HelpData::description::get(),
51  columns - DescStart};
52  it != utility::utf8::line_iterator{};) {
53  stream << *it;
54 
55  // If there's more data to follow, then add the offset
56  if (++it != utility::utf8::line_iterator{}) {
57  stream << '\n' << utility::create_sequence_cts_t<DescStart, ' '>::get();
58  }
59  }
60  }
61 
62  stream << "\n" << reset_colour;
63  }
64  }
65 
66 private:
67  template <std::size_t Depth>
68  [[nodiscard]] constexpr static std::size_t indent_size() noexcept
69  {
70  return Depth * Indent;
71  }
72 };
