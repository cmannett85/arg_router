140  template <typename DependsNodesTuple, typename ModeType>
141  struct cyclic_dependency_checker {
142  // For each depends, find all of its depends, stop when there are no more or if you hit this
143  // policy - static_assert
144  template <std::size_t I, typename Nodes>
145  [[nodiscard]] constexpr static bool check() noexcept
146  {
147  if constexpr (I >= std::tuple_size_v<Nodes>) {
148  return true;
149  } else {
150  using depends_type = std::tuple_element_t<I, Nodes>;
151  if constexpr (algorithm::has_specialisation_v<
152  dependent_t,
153  typename depends_type::policies_type>) {
154  if constexpr (boost::mp11::mp_contains<typename depends_type::policies_type,
155  dependent_t>::value) {
156  return false;
157  } else {
158  using targets =
159  typename depends_targets<typename depends_type::depends_policies_type,
160  ModeType>::type;
161 
162  return check<I + 1, boost::mp11::mp_append<Nodes, targets>>();
163  }
164  }
165 
166  return check<I + 1, Nodes>();
167  }
168  }
