173  template <typename DependsNodesTuple, typename ModeType>
174  struct cyclic_dependency_checker {
175  // For each depends, find all of its depends, stop when there are no
176  // more or if you hit this policy - static_assert
177  template <std::size_t I, typename Nodes>
178  constexpr static bool check()
179  {
180  if constexpr (I >= std::tuple_size_v<Nodes>) {
181  return true;
182  } else {
183  using depends_type = std::tuple_element_t<I, Nodes>;
184  if constexpr (algorithm::has_specialisation_v<
185  dependent_t,
186  typename depends_type::policies_type>) {
187  if constexpr (boost::mp11::mp_contains<
188  typename depends_type::policies_type,
189  dependent_t>::value) {
190  return false;
191  } else {
192  using targets = typename depends_targets<
193  typename depends_type::depends_policies_type,
194  ModeType>::type;
195 
196  return check<I + 1,
197  boost::mp11::mp_append<Nodes, targets>>();
198  }
199  }
200 
201  return check<I + 1, Nodes>();
202  }
203  }
