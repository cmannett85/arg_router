116  // Starting from ModeType, recurse down through the tree and find all the
117  // nodes referred to in DependsPolicies
118  template <typename DependsPoliciesTuple, typename ModeType>
119  struct depends_targets {
120  template <typename Current, typename... Parents>
121  struct visitor {
122  // If current is one of the DependsPolicies and Parents is not
123  // empty, then use the first element of Parents (i.e. the owning
124  // node of the name policy). If not, then set to void
125  using type = boost::mp11::mp_eval_if_c<
126  !(boost::mp11::mp_contains<DependsPoliciesTuple,
127  Current>::value &&
128  (sizeof...(Parents) > 0)),
129  void,

132  traits::integral_constant<std::size_t{0}>>;

149  template <typename DependsNodesTuple, typename ModeType>
150  struct cyclic_dependency_checker {
151  // For each depends, find all of its depends, stop when there are no
152  // more or if you hit this policy - static_assert
153  template <std::size_t I, typename Nodes>
154  [[nodiscard]] constexpr static bool check() noexcept
155  {
156  if constexpr (I >= std::tuple_size_v<Nodes>) {
157  return true;
158  } else {
159  using depends_type = std::tuple_element_t<I, Nodes>;
160  if constexpr (algorithm::has_specialisation_v<
161  dependent_t,
162  typename depends_type::policies_type>) {
163  if constexpr (boost::mp11::mp_contains<
164  typename depends_type::policies_type,
165  dependent_t>::value) {
166  return false;
167  } else {
168  using targets = typename depends_targets<
169  typename depends_type::depends_policies_type,
170  ModeType>::type;
171 
172  return check<I + 1,
173  boost::mp11::mp_append<Nodes, targets>>();
174  }
175  }
176 
177  return check<I + 1, Nodes>();
178  }
179  }
180 
181  constexpr static bool value = check<0, DependsNodesTuple>();
