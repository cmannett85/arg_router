204 template <std::size_t N>
205 [[nodiscard]] constexpr bool GB12_13(
206  const std::array<grapheme_cluster_break_property, N>& trailing_window,
207  grapheme_cluster_break_property next_property) noexcept
208 {
209  static_assert(N > 0, "Trailing window must be at least 1 element");
210 
211  // Count the preceding RIs, if there's an odd number - do not break
212  auto i = 0u;
213  for (; i < trailing_window.size(); ++i) {
214  if (trailing_window[i] != grapheme_cluster_break_property::regional_indicator) {
215  break;
216  }
217  }
218 
219  return ((i % 2) != 0) && (next_property == grapheme_cluster_break_property::regional_indicator);
220 };
