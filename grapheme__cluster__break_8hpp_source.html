199 template <std::size_t N>
200 [[nodiscard]] constexpr bool GB12_13(
201  const std::array<grapheme_cluster_break_class, N>& trailing_window,
202  grapheme_cluster_break_class next_class) noexcept
203 {
204  static_assert(N > 0, "Trailing window must be at least 1 element");
205 
206  // Count the preceding RIs, if there's an odd number - do not break
207  auto i = 0u;
208  for (; i < trailing_window.size(); ++i) {
209  if (trailing_window[i] != grapheme_cluster_break_class::regional_indicator) {
210  break;
211  }
212  }
213 
214  return ((i % 2) != 0) && (next_class == grapheme_cluster_break_class::regional_indicator);
215 };
