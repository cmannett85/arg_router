59  template <typename ProcessedTarget, typename... Parents>
60  [[nodiscard]] parsing::pre_parse_result pre_parse_phase(
61  parsing::dynamic_token_adapter& tokens,
62  [[maybe_unused]] utility::compile_time_optional<ProcessedTarget> processed_target,
63  parsing::parse_target& target,
64  const Parents&... parents) const
65  {
66  using node_type = boost::mp11::mp_first<std::tuple<Parents...>>;
67  static_assert(!node_type::template any_phases_v<typename node_type::value_type,
68  policy::has_parse_phase_method,
69  policy::has_validation_phase_method,
70  policy::has_routing_phase_method>,
71  "Alias owning node cannot have policies that support parse, "
72  "validation, or routing phases");
73 
74  // Find the owning mode
75  using mode_data = policy::nearest_mode_like<std::tuple<Parents...>>;
76  using mode_type = typename mode_data::type;
77  static_assert(!std::is_void_v<mode_type>, "Cannot find parent mode");
78 
79  // Perform all the compile-time checking
80  constexpr auto count = node_fixed_count<node_type>();
81  {
82  using targets = typename alias_targets<aliased_policies_type, mode_type>::type;
83  static_assert(cyclic_dependency_checker<targets, mode_type>::value,
84  "Cyclic dependency detected");
85 
86  // Verify that all the targets have the same count
87  check_target_counts<count, targets>();
88  }
89 
90  // +1 because the node must be named
91  if ((count + 1) > tokens.size()) {
92  return multi_lang_exception{error_code::too_few_values_for_alias,
93  parsing::node_token_type<node_type>()};
94  }
95 
96  // Because this node's job is to create sub-targets, and therefore isn't a 'real' node in
97  // itself, it needs to return skip_node_but_use_sub_targets so the owning tree node keeps
98  // the side-effects (i.e. the sub-targets) but doesn't check the label token. We don't want
99  // label checking in the owning tree_node because we need to _replace_ the alias label
100  // token with the aliased tokens, so any label check against this node will fail. However,
101  // we do need to do a label check _here_, otherwise this aliasing will occur everytime the
102  // pre-parse is called - even on tokens that do not belong to this aliased node!
103  {
104  auto alias_label = *tokens.begin();
105  if (alias_label.prefix == parsing::prefix_type::none) {
106  alias_label = parsing::get_token_type(alias_label.name);
107  }
108 
109  if (!parsing::match<node_type>(alias_label)) {
110  return parsing::pre_parse_action::skip_node;
111  }
112  }
113 
114  // Attempt a transfer so we can guarantee that the original tokens are in the processed
115  // container (this is a no-op if they already are)
116  tokens.transfer(tokens.begin() + count);
117 
118  // Now do the runtime target building
119  const auto visitor = [&](const auto& current, [[maybe_unused]] const auto&... parents) {
120  using policies_type = typename std::decay_t<decltype(current)>::policies_type;
121 
122  using intersection =
123  boost::mp11::mp_set_intersection<aliased_policies_type, policies_type>;
124 
125  if constexpr ((std::tuple_size_v<intersection>) > 0) {
126  auto target_tokens = vector<parsing::token_type>{};
127  target_tokens.reserve(count);
128 
129  auto value_it = tokens.begin();
130  for (auto j = 0u; j < count; ++j) {
131  // Pre-increment the value iterator so we skip over the label token
132  target_tokens.push_back(*(++value_it));
133  }
134 
135  target.add_sub_target({std::move(target_tokens), //
136  current,
137  parents...});
138  }
139  };
140  const auto& mode =
141  std::get<mode_data::index::value>(std::tuple{std::cref(parents)...}).get();
142  utility::tree_recursor(visitor, mode);
143 
144  // Now remove the original tokens as they refer to the aliased node
145  tokens.processed().erase(tokens.processed().begin(),
146  tokens.processed().begin() + count + 1);
147 
148  // The owning node's name checker will fail us now (because we removed this node's label
149  // token), but we still want to keep the processed and unprocessed container changes for
150  // later processing
151  return parsing::pre_parse_action::skip_node_but_use_sub_targets;
152  }

167  template <typename NodeType>
168  [[nodiscard]] constexpr static std::size_t node_fixed_count() noexcept
169  {
170  static_assert(traits::has_minimum_count_method_v<NodeType> &&
171  traits::has_maximum_count_method_v<NodeType>,
172  "Aliased nodes must have minimum and maximum count methods");
173  static_assert(NodeType::minimum_count() == NodeType::maximum_count(),
174  "Aliased nodes must have a fixed count");

176  return NodeType::minimum_count();
177  }

179  // Starting from ModeType, recurse down through the tree and find all the nodes referred to in
180  // AliasPolicies
181  template <typename AliasPoliciesTuple, typename ModeType>
182  struct alias_targets {
183  template <typename Current, typename... Parents>
184  struct visitor {
185  // If current is one of the AliasedPolicies and Parents is not empty, then use the first
186  // element of Parents (i.e. the owning node of the name policy). If not, then set to
187  // void
188  using type = boost::mp11::mp_eval_if_c<
189  !(boost::mp11::mp_contains<AliasPoliciesTuple, Current>::value &&
190  (sizeof...(Parents) > 0)),
191  void,
192  boost::mp11::mp_at,
193  std::tuple<Parents...>,
194  traits::integral_constant<std::size_t{0}>>;
195  };
196 
197  using type =
198  boost::mp11::mp_remove_if<utility::tree_type_recursor_collector_t<visitor, ModeType>,
199  std::is_void>;
200 
201  static_assert(std::tuple_size_v<type> == std::tuple_size_v<AliasPoliciesTuple>,
202  "Number of found modes must match alias policy count");
203  static_assert(std::tuple_size_v<boost::mp11::mp_unique<type>> == std::tuple_size_v<type>,
204  "Node alias list must be unique, do you have short and long "
205  "names from the same node?");
206  };

208  template <typename AliasNodesTuple, typename ModeType>
209  struct cyclic_dependency_checker {
210  // For each alias, find all of its alias, stop when there are no more or if you hit this
211  // policy - static_assert
212  template <std::size_t I, typename Nodes>
213  [[nodiscard]] constexpr static bool check() noexcept
214  {
215  if constexpr (I >= std::tuple_size_v<Nodes>) {
216  return true;
217  } else {
218  using aliased_type = std::tuple_element_t<I, Nodes>;
219  if constexpr (algorithm::has_specialisation_v<
220  alias_t,
221  typename aliased_type::policies_type>) {
222  if constexpr (boost::mp11::mp_contains<typename aliased_type::policies_type,
223  alias_t>::value) {
224  return false;
225  } else {
226  using targets =
227  typename alias_targets<typename aliased_type::aliased_policies_type,
228  ModeType>::type;
229 
230  return check<I + 1, boost::mp11::mp_append<Nodes, targets>>();
231  }
232  }
233 
234  return check<I + 1, Nodes>();
235  }
236  }
237 
238  constexpr static bool value = check<0, AliasNodesTuple>();
239  };

244  utility::tuple_type_iterator<TargetsTuple>([](auto i) {
245  using target_type = std::tuple_element_t<i, TargetsTuple>;
246 
247  static_assert(node_fixed_count<target_type>() == Count,
248  "All alias targets must have a count that matches the owner");
249  });
250  }
