204  // For each alias, find all of its alias, stop when there are no
205  // more or if you hit this policy - static_assert
206  template <std::size_t I, typename Nodes>
207  constexpr static bool check()
208  {
209  if constexpr (I >= std::tuple_size_v<Nodes>) {
210  return true;
211  } else {
212  using aliased_type = std::tuple_element_t<I, Nodes>;
213  if constexpr (algorithm::has_specialisation_v<
214  alias_t,
215  typename aliased_type::policies_type>) {
216  if constexpr (boost::mp11::mp_contains<
217  typename aliased_type::policies_type,
218  alias_t>::value) {
219  return false;
220  } else {
221  using targets = typename alias_targets<
222  typename aliased_type::aliased_policies_type,
223  ModeType>::type;
224 
225  return check<I + 1,
226  boost::mp11::mp_append<Nodes, targets>>();
227  }
228  }
229

231  }
