257 template <template <typename...> typename Fn, typename U>
258 [[nodiscard]] constexpr auto tuple_filter_and_construct(U&& input) noexcept
259 {
260  // Zip together an index-sequence of U and the elements of it
261  using zipped =
262  zip_t<boost::mp11::mp_iota_c<std::tuple_size_v<std::decay_t<U>>>,
263  std::decay_t<U>>;
264 
265  // Filter out types not approved by Fn, we need to wrap Fn so that it only
266  // operates on the second type in the pair (the type from U)
267  using wrapped_fn = boost::mp11::mp_bind<
268  Fn,
269  boost::mp11::mp_bind<boost::mp11::mp_second, boost::mp11::_1>>;
270  using filtered = boost::mp11::mp_filter<wrapped_fn::template fn, zipped>;
271 
272  // Fold construct the return value using the zip indices as accessors to
273  // input
274  return detail::tuple_filter_and_construct_impl(
275  input,
276  typename unzip<filtered>::first_type{});
277 }
