358 template <std::size_t N, typename T>
359 constexpr T remove_bit(T value)
360 {
361  static_assert(std::is_integral_v<T> && std::is_unsigned_v<T>,
362  "T must be an unsigned integral");
363  static_assert(N < (sizeof(T) * 8),
364  "N is larger than the number of bit available");
365 
366  // Store the bits lower than N
367  const auto prefix = value & T{(1 << N) - 1};
368  value >>= N + 1;
369  value <<= N;
370  return value | prefix;
