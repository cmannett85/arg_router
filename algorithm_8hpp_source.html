297 template <template <typename...> typename Fn, typename U>
298 [[nodiscard]] constexpr auto tuple_filter_and_construct(U&& input) noexcept
299 {
300  // Zip together an index-sequence of U and the elements of it
301  using zipped =
302  zip_t<boost::mp11::mp_iota_c<std::tuple_size_v<std::decay_t<U>>>,
303  std::decay_t<U>>;
304 
305  // Filter out types not approved by Fn, we need to wrap Fn so that it only
306  // operates on the second type in the pair (the type from U)
307  using wrapped_fn = boost::mp11::mp_bind<
308  Fn,
309  boost::mp11::mp_bind<boost::mp11::mp_second, boost::mp11::_1>>;
310  using filtered = boost::mp11::mp_filter<wrapped_fn::template fn, zipped>;
311 
312  // Fold construct the return value using the zip indices as accessors to
313  // input
314  return detail::tuple_filter_and_construct_impl(
315  input,
316  typename unzip<filtered>::first_type{});
317 }
