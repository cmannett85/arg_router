132  void parse(parsing::token_list& tokens, const Parents&... parents) const
133  {
134  // Remove our token (if not anonymous) from the list, keep it for any
135  // error messages
136  auto mode_token = std::optional<parsing::token_type>{};
137  if constexpr (!is_anonymous) {
138  mode_token = tokens.front();
139  tokens.erase(tokens.begin());
140  }

142  // Find any matching mode type and delegate parsing into it
143  auto found_child_mode = false;
144  if (!is_anonymous && !tokens.empty()) {
145  tree_node<Params...>::find(
146  tokens.front(),
147  [&](auto /*i*/, const auto& child) {

161  // optional-wrapped children's value_types but any no_result_value

171  const auto found = tree_node<Params...>::find(
172  tokens.front(),

183  }

189  }

192  utility::tuple_iterator(
193  [&](auto i, auto& result) {

198  }

205  if constexpr (!std::is_void_v<routing_policy>) {
206  // Strip out the skipped results

211  std::apply(

218  } else if constexpr (is_anonymous) {
