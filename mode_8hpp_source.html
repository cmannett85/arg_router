135  template <typename Validator, bool HasTarget, typename... Parents>
136  [[nodiscard]] std::optional<parsing::parse_target> pre_parse(
137  parsing::pre_parse_data<Validator, HasTarget> pre_parse_data,
138  const Parents&... parents) const
139  {
140  using namespace utility::string_view_ops;
141  using namespace std::string_literals;
142 
143  static_assert(!HasTarget,
144  "Modes cannot receive pre_parse_data containing parent "
145  "parse_targets");
146  static_assert(!is_anonymous || (sizeof...(Parents) <= 1),
147  "Anonymous modes can only exist under the root");
148 
149  auto& args = pre_parse_data.args();
150 
151  // If we're not anonymous then check the leading token is a match. We can just delegate to
152  // the default implementation for this
153  if constexpr (!is_anonymous) {
154  auto match = parent_type::pre_parse(pre_parse_data, *this, parents...);
155  if (!match) {
156  return match;
157  }
158 
159  // Check if the next token (if any) matches a child mode. If so, delegate to that
160  if (!args.empty()) {
161  match.reset();
162  utility::tuple_iterator(
163  [&](auto /*i*/, const auto& child) {
164  using child_type = std::decay_t<decltype(child)>;
165  if constexpr (traits::is_specialisation_of_v<child_type, mode_t>) {
166  if (!match) {
167  match = child.pre_parse(pre_parse_data, *this, parents...);
168  }
169  }
170  },
171  this->children());
172  if (match) {
173  return match;
174  }
175  }
176  }
177 
178  if (!pre_parse_data.validator()(*this, parents...)) {
179  return {};
180  }
181 
182  auto target = parsing::parse_target{*this, parents...};
183 
184  // Iterate over the tokens until consumed, skipping children already processed that cannot
185  // be repeated on the command line
186  auto matched = std::bitset<std::tuple_size_v<children_type>>{};
187  while (!args.empty()) {
188  // Take a copy of the front token for the error messages
189  const auto front_token = args.front();
190 
191  auto match = std::optional<parsing::parse_target>{};
192  utility::tuple_iterator(
193  [&]([[maybe_unused]] auto i, const auto& child) {
194  using child_type = std::decay_t<decltype(child)>;
195 
196  // The token(s) have been processed so skip over any remaining children
197  if (match) {
198  return;
199  }
200 
201  // Skip past modes, as they're handled earlier
202  if constexpr (!traits::is_specialisation_of_v<child_type, mode_t>) {
203  match = child.pre_parse(
204  parsing::pre_parse_data{
205  args,
206  target,
207  [&](const auto& real_child, const auto&...) {
208  using real_child_type = std::decay_t<decltype(real_child)>;
209  return verify_match<real_child_type>(matched[i], front_token);
210  }},
211  *this,
212  parents...);
213 
214  // Update the matched bitset
215  if (match) {
216  matched.set(i);
217  }
218  }
219  },
220  this->children());
221 
222  if (!match) {
223  if (matched.all()) {
224  throw multi_lang_exception{error_code::unhandled_arguments, args};
225  }
226 
227  throw multi_lang_exception{error_code::unknown_argument, front_token};
228  }
229 
230  // Flatten out nested sub-targets
231  if (match->sub_targets().empty()) {
232  target.add_sub_target(std::move(*match));
233  } else {
234  for (auto& sub_target : match->sub_targets()) {
235  target.add_sub_target(std::move(sub_target));
236  }
237  }
238  }
239 
240  return target;
241  }

253  template <typename... Parents>
254  void parse(parsing::parse_target target, const Parents&... parents) const
255  {
256  // Create an internal-use results tuple, this is made from optional-wrapped children's
257  // value_types but any no_result_value children are replaced with skip_tag - this makes it
258  // easier to iterate
259  using results_type = boost::mp11::mp_transform<optional_value_type_or_skip, children_type>;
260  auto results = results_type{};
261 
262  for (auto& sub_target : target.sub_targets()) {
263  const auto node_hash = sub_target.node_type();
264  auto result = sub_target();
265 
266  if (result.has_value()) {
267  // We need to find the matching node to the sub_target. We do this by searching
268  // each child's subtree for a match - this gives us the index into the results tuple
269  // and the sub node type for the match
270  auto found = false;
271  utility::tuple_iterator(
272  [&](auto i, const auto& child) {
273  if (found) {
274  return;
275  }
276 
277  match_child(child, node_hash, [&](const auto& sub_child) {
278  found = true;
279  process_result<i>(results, result, sub_child);
280  });
281  },
282  this->children());
283  }
284  }
285 
286  // Handle missing tokens
287  utility::tuple_iterator(
288  [&]([[maybe_unused]] auto i, auto& result) {
289  if constexpr (!is_skip_tag_v<std::decay_t<decltype(result)>>) {
290  if (!result) {
291  const auto& child = std::get<i>(this->children());
292  process_missing_token(result, child, *this, parents...);
293  }
294  }
295  },
296  results);
297 
298  // Handle multi-stage value validation. Multi-stage value nodes cannot be validated during
299  // processing, as they will likely fail validation when partially processed. So the
300  // multi-stage nodes do not perform validation themselves, but have their owner (i.e. modes)
301  // do it at the end of processing - including after any default values have been generated
302  multi_stage_validation(results, *this, parents...);
303 
304  // Routing
305  using routing_policy =
306  typename parent_type::template phase_finder_t<policy::has_routing_phase_method>;
307  if constexpr (!std::is_void_v<routing_policy>) {
308  // Strip out the skipped results
309  auto stripped_results = algorithm::tuple_filter_and_construct<
310  boost::mp11::mp_not_fn<is_skip_tag>::template fn>(std::move(results));
311 
312  std::apply(
313  [&](auto&&... args) {
314  this->routing_phase(std::forward<std::decay_t<decltype(*args)>>(*args)...);
315  },
316  std::move(stripped_results));
317  } else if constexpr (is_anonymous) {
318  static_assert(traits::always_false_v<Params...>, "Anonymous modes must have routing");
319  } else if constexpr (!boost::mp11::mp_all_of<children_type, is_child_mode>::value) {
320  static_assert(traits::always_false_v<Params...>,
321  "Mode must have a router or all its children are modes");
322  } else {
323  // Mode guaranteed to be named here
324  throw multi_lang_exception{error_code::mode_requires_arguments,
325  parsing::node_token_type<mode_t>()};
326  }
327  }

351  template <typename Child, typename Handler>
352  static void match_child(const Child& child, std::size_t hash, Handler handler)
353  {
354  auto found = false;
355  utility::tree_recursor(
356  [&](const auto& node, const auto&...) {
357  using node_type = std::decay_t<decltype(node)>;
358 
359  // Skip modes and nodes without a parse function. In reality the runtime code will
360  // never allow that, but let's help the compiler out...
361  if constexpr (!traits::is_specialisation_of_v<node_type, mode_t> &&
362  traits::has_parse_method_v<node_type>) {
363  if (!found && (utility::type_hash<node_type>() == hash)) {
364  found = true;
365  handler(node);
366  }
367  }
368  },
369  child);

372  template <typename Child>
373  [[nodiscard]] static bool verify_match(bool already_matched,
374  [[maybe_unused]] parsing::token_type token)
375  {
376  if constexpr (!Child::is_named && !policy::has_multi_stage_value_v<Child>) {
377  // Child is not named and can only appear on the command line once, so only perform the
378  // pre-parse if it hasn't been matched already
379  return !already_matched;
380  } else if constexpr (Child::is_named && !policy::has_multi_stage_value_v<Child>) {
381  // Child is named, but can only appear once on the command line, so perform the
382  // pre-parse and if there is a match check it isn't already matched
383  if (already_matched) {
384  throw multi_lang_exception{error_code::argument_has_already_been_set, token};
385  }
386  }
387 
388  return true;
389  }

391  template <std::size_t I, typename ResultsType, typename ChildType>
392  void process_result(ResultsType& results,
393  utility::unsafe_any parse_result,
394  const ChildType& child) const
395  {
396  using optional_result_type = std::tuple_element_t<I, ResultsType>;
397 
398  // The mode check is just for the compiler, the runtime code prevents it reaching here
399  if constexpr (!is_skip_tag_v<optional_result_type>) {
400  auto& result = std::get<I>(results);
401 
402  if constexpr (policy::has_multi_stage_value_v<ChildType>) {
403  using result_type = traits::arg_type_at_index<decltype(&ChildType::merge), 1>;
404  child.merge(result, std::move(parse_result.get<result_type>()));
405  } else {
406  if (result) {
407  throw multi_lang_exception{error_code::argument_has_already_been_set,
408  parsing::node_token_type<ChildType>()};
409  }
410 
411  using result_type = decltype(std::declval<ChildType>().parse(
412  std::declval<parsing::parse_target>()));
413  result = std::move(parse_result.get<result_type>());
414  }
415  }
416  }

418  template <typename ValueType, typename ChildType, typename... Parents>
419  void process_missing_token(std::optional<ValueType>& result,
420  const ChildType& child,
421  const Parents&... parents) const
422  {
423  utility::tuple_type_iterator<typename ChildType::policies_type>([&](auto i) {
424  using policy_type = std::tuple_element_t<i, typename ChildType::policies_type>;
425  if constexpr (policy::has_missing_phase_method_v<policy_type, ValueType>) {
426  result = child.policy_type::template missing_phase<ValueType>(child, parents...);
427  }
428  });
429 
430  // If no missing_phase methods were found that made the result valid, then it still needs to
431  // be valid - just default initialise
432  if (!result) {
433  result = ValueType{};
434  }
435 
436  // Irritatingly, we have to run the validation phase on the new value
437  utility::tuple_type_iterator<typename ChildType::policies_type>([&](auto i) {
438  using policy_type = std::tuple_element_t<i, typename ChildType::policies_type>;
439  if constexpr (policy::has_validation_phase_method_v<policy_type, ValueType>) {
440  child.policy_type::template validation_phase(*result, child, parents...);
441  }
442  });
443  }

445  template <typename ResultsType, typename... Parents>
446  void multi_stage_validation(const ResultsType& results, const Parents&... parents) const
447  {
448  // For all the children that use multi-stage values, invoke any policy
449  // that supports a validation phase
450  utility::tuple_iterator(
451  [&](auto i, const auto& child) {
452  using child_type = std::decay_t<decltype(child)>;
453  using child_policies_type = typename child_type::policies_type;
454  using optional_result_type = std::tuple_element_t<i, ResultsType>;
455 
456  constexpr auto msv = policy::has_multi_stage_value_v<child_type> ||
457  boost::mp11::mp_any_of<typename child_type::children_type,
458  policy::has_multi_stage_value>::value;
459 
460  if constexpr (!is_skip_tag_v<optional_result_type> && msv) {
461  using result_type = typename optional_result_type::value_type;
462  const auto& result = *std::get<i>(results);
463 
464  utility::tuple_type_iterator<child_policies_type>([&](auto j) {
465  using policy_type = std::tuple_element_t<j, child_policies_type>;
466  if constexpr (policy::has_validation_phase_method_v<policy_type,
467  result_type>) {
468  child.policy_type::template validation_phase(result, child, parents...);
469  }
470  });
471  }
472  },
473  this->children());
474  }
