139  template <typename... Parents>
140  void parse(parsing::token_list& tokens, const Parents&... parents) const
141  {
142  static_assert(!is_anonymous || (sizeof...(Parents) <= 1),
143  "Anonymous modes can only exist under the root");
144 
145  // Remove our token (if not anonymous) from the list, keep it for any
146  // error messages
147  auto mode_token = std::optional<parsing::token_type>{};
148  if constexpr (!is_anonymous) {
149  mode_token = tokens.pending_view().front();
150  tokens.mark_as_processed();
151  }
152 
153  // Find any matching mode type and delegate parsing into it
154  if constexpr (!is_anonymous) {
155  if (!tokens.pending_view().empty()) {
156  auto found_child_mode = false;
157  parent_type::find(
158  tokens.pending_view().front(),
159  [&](auto /*i*/, const auto& child) {
160  using child_type = std::decay_t<decltype(child)>;
161  if constexpr (traits::is_specialisation_of_v<child_type,
162  mode_t>) {
163  found_child_mode = true;
164  child.parse(tokens, *this, parents...);
165  }
166  });
167  if (found_child_mode) {
168  return;
169  }
170  }
171  }
172 
173  // Create an internal-use results tuple, this is made from
174  // optional-wrapped children's value_types but any no_result_value
175  // children are replaced with skip_tag - this makes it easier to iterate
176  using results_type =
177  boost::mp11::mp_transform<optional_value_type_or_skip,
178  children_type>;
179  auto results = results_type{};
180 
181  // Iterate over the tokens until consumed
182  while (!tokens.pending_view().empty()) {
183  // Find the matching child
184  const auto found = parent_type::find(
185  tokens.pending_view().front(),
186  [&](auto i, const auto& child) {
187  process_result<i.value>(tokens,
188  results,
189  child,
190  *this,
191  parents...);
192  },
193  results);
194  if (!found) {
195  throw parse_exception{"Unknown argument",
196  tokens.pending_view().front()};
197  }
198  }
199 
200  // Handle missing tokens
201  utility::tuple_iterator(
202  [&](auto i, auto& result) {
203  if constexpr (!is_skip_tag_v<std::decay_t<decltype(result)>>) {
204  if (!result) {
205  const auto& child = std::get<i>(this->children());
206  process_missing_token(result, child, *this, parents...);
207  }
208  }
209  },
210  results);
211 
212  // Handle multi-stage value validation. Multi-stage value nodes cannot
213  // be validated during processing, as they will likely fail validation
214  // when partially processed. So the multi-stage nodes do not validation
215  // themselves, but have their owner (i.e. modes) do it at the end of
216  // processing - including after any default values have been generated
217  multi_stage_validation(results, *this, parents...);
218 
219  // Routing
220  using routing_policy = typename parent_type::template phase_finder_t<
221  policy::has_routing_phase_method>;
222  if constexpr (!std::is_void_v<routing_policy>) {
223  // Strip out the skipped results
224  auto stripped_results = algorithm::tuple_filter_and_construct<
225  boost::mp11::mp_not_fn<is_skip_tag>::template fn>(
226  std::move(results));
227 
228  std::apply(
229  [&](auto&&... args) {
230  this->routing_policy::routing_phase(
231  tokens, // Will always be empty
232  std::forward<std::decay_t<decltype(*args)>>(*args)...);
233  },
234  std::move(stripped_results));
235  } else if constexpr (is_anonymous) {
236  static_assert(traits::always_false_v<Params...>,
237  "Anonymous modes must have routing");
238  } else if constexpr (!boost::mp11::mp_all_of<children_type,
239  is_child_mode>::value) {
240  static_assert(
241  traits::always_false_v<Params...>,
242  "Mode must have a router or all its children are modes");
243  } else {
244  // mode_token guaranteed to be valid here, because it the mode must
245  // not be anonymous to reach here
246  throw parse_exception{"Mode requires arguments", *mode_token};
247  }
248  }

277  template <std::size_t I,
278  typename ResultsType,
279  typename ChildType,
280  typename... Parents>
281  void process_result(parsing::token_list& tokens,
282  ResultsType& results,
283  const ChildType& child,
284  const Parents&... parents) const
285  {
286  using optional_result_type = std::tuple_element_t<I, ResultsType>;
287 
288  if constexpr (!is_skip_tag_v<optional_result_type>) {
289  // Make a copy of the token in case we need it for the error message
290  // later (parsing consumes the tokens)
291  const auto first_token = tokens.pending_view().front();
292 
293  auto parse_result = child.parse(tokens, parents...);
294  auto& result = std::get<I>(results);
295 
296  if constexpr (policy::has_multi_stage_value_v<ChildType>) {
297  child.merge(result, std::move(parse_result));
298  } else {
299  if (result) {
300  throw parse_exception{"Argument has already been set",
301  first_token};
302  }
303 
304  result = std::move(parse_result);
305  }
306  } else {
307  // Even if the skip tag is set, we still need to process the token
308  // as it may have side effects that affect other token processing
309  // (e.g. aliases)
310  child.parse(tokens, parents...);
311  }
312  }

314  template <typename ValueType, typename ChildType, typename... Parents>
315  void process_missing_token(std::optional<ValueType>& result,
316  const ChildType& child,
317  const Parents&... parents) const
318  {
319  utility::tuple_type_iterator<typename ChildType::policies_type>(
320  [&](auto i) {
321  using policy_type =
322  std::tuple_element_t<i, typename ChildType::policies_type>;
323  if constexpr (policy::has_missing_phase_method_v<policy_type,
324  ValueType>) {
325  result =
326  child.policy_type::template missing_phase<ValueType>(
327  child,
328  parents...);
329  }
330  });
331 
332  // If no missing_phase methods were found that made the result valid,
333  // then it still needs to be valid - just default initialise
334  if (!result) {
335  result = ValueType{};
336  }
337 
338  // Irritatingly, we have to run the validation phase on the new value
339  utility::tuple_type_iterator<
340  typename ChildType::policies_type>([&](auto i) {
341  using policy_type =
342  std::tuple_element_t<i, typename ChildType::policies_type>;
343  if constexpr (policy::has_validation_phase_method_v<policy_type,
344  ValueType>) {
345  child.policy_type::template validation_phase(*result,
346  child,
347  parents...);
348  }
349  });
350  }

352  template <typename ResultsType, typename... Parents>
353  void multi_stage_validation(const ResultsType& results,
354  const Parents&... parents) const
355  {
356  // For all the children that use multi-stage values, invoke any policy
357  // that supports a validation phase
358  utility::tuple_iterator(
359  [&](auto i, const auto& child) {
360  using child_type = std::decay_t<decltype(child)>;
361  using child_policies_type = typename child_type::policies_type;
362  using optional_result_type =
363  std::tuple_element_t<i, ResultsType>;
364 
365  if constexpr (!is_skip_tag_v<optional_result_type> &&
366  policy::has_multi_stage_value_v<child_type>) {
367  using result_type =
368  typename optional_result_type::value_type;
369  const auto& result = *std::get<i>(results);
370 
371  utility::tuple_type_iterator<child_policies_type>(
372  [&](auto j) {
373  using policy_type =
374  std::tuple_element_t<j, child_policies_type>;
375  if constexpr (policy::has_validation_phase_method_v<
376  policy_type,
377  result_type>) {
378  child.policy_type::template validation_phase(
379  result,
380  child,
381  parents...);
382  }
383  });
384  }
385  },
386  this->children());
387  }
