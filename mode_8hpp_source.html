114  template <typename Validator, bool HasTarget, typename... Parents>
115  [[nodiscard]] std::optional<parsing::parse_target> pre_parse(
116  parsing::pre_parse_data<Validator, HasTarget> pre_parse_data,
117  const Parents&... parents) const
118  {
119  using namespace utility::string_view_ops;
120  using namespace std::string_literals;
121 
122  static_assert(!HasTarget,
123  "Modes cannot receive pre_parse_data containing parent "
124  "parse_targets");
125  static_assert(!is_anonymous || (sizeof...(Parents) <= 1),
126  "Anonymous modes can only exist under the root");
127 
128  auto& args = pre_parse_data.args();
129 
130  // If we're not anonymous then check the leading token is a match. We can just delegate to
131  // the default implementation for this
132  if constexpr (!is_anonymous) {
133  auto match = parent_type::pre_parse(pre_parse_data, *this, parents...);
134  if (!match) {
135  return match;
136  }
137 
138  // Check if the next token (if any) matches a child mode. If so, delegate to that
139  if (!args.empty()) {
140  match.reset();
141  utility::tuple_iterator(
142  [&](auto /*i*/, const auto& child) {
143  using child_type = std::decay_t<decltype(child)>;
144  if constexpr (traits::is_specialisation_of_v<child_type, mode_t>) {
145  if (!match) {
146  match = child.pre_parse(pre_parse_data, *this, parents...);
147  }
148  }
149  },
150  this->children());
151  if (match) {
152  return match;
153  }
154  }
155  }
156 
157  if (!pre_parse_data.validator()(*this, parents...)) {
158  return {};
159  }
160 
161  auto target = parsing::parse_target{*this, parents...};
162 
163  // Iterate over the tokens until consumed, skipping children already processed that cannot
164  // be repeated on the command line
165  auto matched = std::bitset<std::tuple_size_v<children_type>>{};
166  while (!args.empty()) {
167  // Take a copy of the front token for the error messages
168  const auto front_token = args.front();
169 
170  auto match = std::optional<parsing::parse_target>{};
171  utility::tuple_iterator(
172  [&]([[maybe_unused]] auto i, const auto& child) {
173  using child_type = std::decay_t<decltype(child)>;
174 
175  // The token(s) have been processed so skip over any remaining children
176  if (match) {
177  return;
178  }
179 
180  // Skip past modes, as they're handled earlier
181  if constexpr (!traits::is_specialisation_of_v<child_type, mode_t>) {
182  match = child.pre_parse(
183  parsing::pre_parse_data{
184  args,
185  target,
186  [&](const auto& real_child, const auto&...) {
187  using real_child_type = std::decay_t<decltype(real_child)>;
188  return verify_match<real_child_type>(matched[i], front_token);
189  }},
190  *this,
191  parents...);
192 
193  // Update the matched bitset
194  if (match) {
195  matched.set(i);
196  }
197  }
198  },
199  this->children());
200 
201  if (!match) {
202  if (matched.all()) {
203  throw parse_exception{"Unhandled arguments", args};
204  }
205 
206  throw parse_exception{"Unknown argument", front_token};
207  }
208 
209  // Flatten out nested sub-targets
210  if (match->sub_targets().empty()) {
211  target.add_sub_target(std::move(*match));
212  } else {
213  for (auto& sub_target : match->sub_targets()) {
214  target.add_sub_target(std::move(sub_target));
215  }
216  }
217  }
218 
219  return target;
220  }

232  template <typename... Parents>
233  void parse(parsing::parse_target target, const Parents&... parents) const
234  {
235  // Create an internal-use results tuple, this is made from optional-wrapped children's
236  // value_types but any no_result_value children are replaced with skip_tag - this makes it
237  // easier to iterate
238  using results_type = boost::mp11::mp_transform<optional_value_type_or_skip, children_type>;
239  auto results = results_type{};
240 
241  for (auto& sub_target : target.sub_targets()) {
242  const auto node_hash = sub_target.node_type();
243  auto result = sub_target();
244 
245  if (result.has_value()) {
246  // We need to find the matching node to the sub_target. We do this by searching
247  // each child's subtree for a match - this gives us the index into the results tuple
248  // and the sub node type for the match
249  auto found = false;
250  utility::tuple_iterator(
251  [&](auto i, const auto& child) {
252  if (found) {
253  return;
254  }
255 
256  match_child(child, node_hash, [&](const auto& sub_child) {
257  found = true;
258  process_result<i>(results, result, sub_child);
259  });
260  },
261  this->children());
262  }
263  }
264 
265  // Handle missing tokens
266  utility::tuple_iterator(
267  [&]([[maybe_unused]] auto i, auto& result) {
268  if constexpr (!is_skip_tag_v<std::decay_t<decltype(result)>>) {
269  if (!result) {
270  const auto& child = std::get<i>(this->children());
271  process_missing_token(result, child, *this, parents...);
272  }
273  }
274  },
275  results);
276 
277  // Handle multi-stage value validation. Multi-stage value nodes cannot be validated during
278  // processing, as they will likely fail validation when partially processed. So the
279  // multi-stage nodes do not perform validation themselves, but have their owner (i.e. modes)
280  // do it at the end of processing - including after any default values have been generated
281  multi_stage_validation(results, *this, parents...);
282 
283  // Routing
284  using routing_policy =
285  typename parent_type::template phase_finder_t<policy::has_routing_phase_method>;
286  if constexpr (!std::is_void_v<routing_policy>) {
287  // Strip out the skipped results
288  auto stripped_results = algorithm::tuple_filter_and_construct<
289  boost::mp11::mp_not_fn<is_skip_tag>::template fn>(std::move(results));
290 
291  std::apply(
292  [&](auto&&... args) {
293  this->routing_policy::routing_phase(
294  std::forward<std::decay_t<decltype(*args)>>(*args)...);
295  },
296  std::move(stripped_results));
297  } else if constexpr (is_anonymous) {
298  static_assert(traits::always_false_v<Params...>, "Anonymous modes must have routing");
299  } else if constexpr (!boost::mp11::mp_all_of<children_type, is_child_mode>::value) {
300  static_assert(traits::always_false_v<Params...>,
301  "Mode must have a router or all its children are modes");
302  } else {
303  // Mode guaranteed to be named here
304  throw parse_exception{"Mode requires arguments", parsing::node_token_type<mode_t>()};
305  }
306  }

330  template <typename Child, typename Handler>
331  static void match_child(const Child& child, std::size_t hash, Handler handler)
332  {
333  auto found = false;
334  utility::tree_recursor(
335  [&](const auto& node, const auto&...) {
336  using node_type = std::decay_t<decltype(node)>;
337 
338  // Skip modes and nodes without a parse function. In reality the runtime code will
339  // never allow that, but let's help the compiler out...
340  if constexpr (!traits::is_specialisation_of_v<node_type, mode_t> &&
341  traits::has_parse_method_v<node_type>) {
342  if (!found && (utility::type_hash<node_type>() == hash)) {
343  found = true;
344  handler(node);
345  }
346  }
347  },
348  child);

351  template <typename Child>
352  static bool verify_match(bool already_matched, [[maybe_unused]] parsing::token_type token)
353  {
354  if constexpr (!Child::is_named && !policy::has_multi_stage_value_v<Child>) {
355  // Child is not named and can only appear on the command line once, so only perform the
356  // pre-parse if it hasn't been matched already
357  return !already_matched;
358  } else if constexpr (Child::is_named && !policy::has_multi_stage_value_v<Child>) {
359  // Child is named, but can only appear once on the command line, so perform the
360  // pre-parse and if there is a match check it isn't already matched
361  if (already_matched) {
362  throw parse_exception{"Argument has already been set", token};
363  }
364  }
365 
366  return true;
367  }

369  template <std::size_t I, typename ResultsType, typename ChildType>
370  void process_result(ResultsType& results,
371  utility::unsafe_any parse_result,
372  const ChildType& child) const
373  {
374  using optional_result_type = std::tuple_element_t<I, ResultsType>;
375 
376  // The mode check is just for the compiler, the runtime code prevents it reaching here
377  if constexpr (!is_skip_tag_v<optional_result_type>) {
378  auto& result = std::get<I>(results);
379 
380  if constexpr (policy::has_multi_stage_value_v<ChildType>) {
381  using result_type = traits::arg_type_at_index<decltype(&ChildType::merge), 1>;
382  child.merge(result, std::move(parse_result.get<result_type>()));
383  } else {
384  if (result) {
385  throw parse_exception{"Argument has already been set",
386  parsing::node_token_type<ChildType>()};
387  }
388 
389  using result_type = decltype(std::declval<ChildType>().parse(
390  std::declval<parsing::parse_target>()));
391  result = std::move(parse_result.get<result_type>());
392  }
393  }

396  template <typename ValueType, typename ChildType, typename... Parents>
397  void process_missing_token(std::optional<ValueType>& result,
398  const ChildType& child,
399  const Parents&... parents) const
400  {
401  utility::tuple_type_iterator<typename ChildType::policies_type>([&](auto i) {
402  using policy_type = std::tuple_element_t<i, typename ChildType::policies_type>;
403  if constexpr (policy::has_missing_phase_method_v<policy_type, ValueType>) {
404  result = child.policy_type::template missing_phase<ValueType>(child, parents...);
405  }
406  });
407 
408  // If no missing_phase methods were found that made the result valid, then it still needs to
409  // be valid - just default initialise
410  if (!result) {
411  result = ValueType{};
412  }
413 
414  // Irritatingly, we have to run the validation phase on the new value
415  utility::tuple_type_iterator<typename ChildType::policies_type>([&](auto i) {
416  using policy_type = std::tuple_element_t<i, typename ChildType::policies_type>;
417  if constexpr (policy::has_validation_phase_method_v<policy_type, ValueType>) {
418  child.policy_type::template validation_phase(*result, child, parents...);
419  }
420  });
421  }

423  template <typename ResultsType, typename... Parents>
424  void multi_stage_validation(const ResultsType& results, const Parents&... parents) const
425  {
426  // For all the children that use multi-stage values, invoke any policy
427  // that supports a validation phase
428  utility::tuple_iterator(
429  [&](auto i, const auto& child) {
430  using child_type = std::decay_t<decltype(child)>;
431  using child_policies_type = typename child_type::policies_type;
432  using optional_result_type = std::tuple_element_t<i, ResultsType>;
433 
434  constexpr auto msv = policy::has_multi_stage_value_v<child_type> ||
435  boost::mp11::mp_any_of<typename child_type::children_type,
436  policy::has_multi_stage_value>::value;
437 
438  if constexpr (!is_skip_tag_v<optional_result_type> && msv) {
439  using result_type = typename optional_result_type::value_type;
440  const auto& result = *std::get<i>(results);
441 
442  utility::tuple_type_iterator<child_policies_type>([&](auto j) {
443  using policy_type = std::tuple_element_t<j, child_policies_type>;
444  if constexpr (policy::has_validation_phase_method_v<policy_type,
445  result_type>) {
446  child.policy_type::template validation_phase(result, child, parents...);
447  }
448  });
449  }
450  },
451  this->children());
452  }
