156  template <typename Fn, typename ResultsTuple = std::tuple<>>
157  constexpr bool find(const parsing::token_type& token,
158  const Fn& visitor,
159  const ResultsTuple& results_tuple = {}) const
160  {
161  auto result = false;
162  utility::tuple_iterator(
163  [&](auto i, const auto& node) {
164  using node_type = std::decay_t<decltype(node)>;
165  constexpr auto match_arity =
166  traits::arity_v<decltype(&node_type::template match<Fn>)>;
167 
168  if (result) {
169  return;
170  }
171 
172  // Wrap the caller's visitor with one that forwards the child
173  // index
174  const auto wrapped_visitor = [&](const auto& child) {
175  visitor(i, child);
176  };
177 
178  if constexpr ((i <= std::tuple_size_v<ResultsTuple>)&& //
179  (match_arity == 3)) {
180  if (node.match(token,
181  wrapped_visitor,
182  std::get<i.value>(results_tuple))) {
183  result = true;
184  }
185  } else if constexpr (match_arity == 2) {
186  if (node.match(token, wrapped_visitor)) {
187  result = true;
188  }
189  }
190  },
191  children());
192 
193  return result;
194  }
