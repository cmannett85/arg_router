201  template <typename Fn, typename ResultsTuple = std::tuple<>>
202  constexpr bool find(parsing::token_type token,
203  const Fn& visitor,
204  const ResultsTuple& results_tuple = {}) const
205  {
206  auto result = false;
207  utility::tuple_iterator(
208  [&](auto i, const auto& node) {
209  if (result) {
210  return;
211  }
212 
213  using node_type = std::decay_t<decltype(node)>;
214 
215  // Wrap the caller's visitor with one that forwards the child
216  // index
217  const auto wrapped_visitor = [&](const auto& child) {
218  visitor(i, child);
219  };
220 
221  constexpr auto match_arity = traits::arity_v<
222  decltype(&node_type::template match<
223  std::decay_t<decltype(wrapped_visitor)>>)>;
224 
225  if constexpr ((i < std::tuple_size_v<ResultsTuple>)&& //
226  (match_arity == 3)) {
227  if (node.match(token,
228  wrapped_visitor,
229  std::get<i.value>(results_tuple))) {
230  result = true;
231  }
232  } else {
233  if (node.match(token, wrapped_visitor)) {
234  result = true;
235  }
236  }
237  },
238  children());
239 
240  return result;

470  template <typename... Parents>
471  [[nodiscard]] bool pre_parse(vector<parsing::token_type>& args,
472  parsing::token_list& tokens,
473  const Parents&... parents) const
474  {
475  auto result = vector<parsing::token_type>{};
476  auto tmp_args = args;
477  auto adapter = parsing::dynamic_token_adapter{result, tmp_args};
478 
479  auto match =
480  parsing::pre_parse_result{parsing::pre_parse_action::valid_node};
481  utility::tuple_type_iterator<
482  priority_ordered_policies_type>([&](auto i) {
483  using policy_type =
484  std::tuple_element_t<i, priority_ordered_policies_type>;
485  if constexpr (policy::has_pre_parse_phase_method_v<policy_type>) {
486  if (match == parsing::pre_parse_action::skip_node) {
487  return;
488  }
489 
490  // Keep the skip_node_but_use_tokens state if later policies
491  // return valid_node
492  const auto keep_tokens =
493  match ==
494  parsing::pre_parse_action::skip_node_but_use_tokens;
495  match =
496  this->policy_type::pre_parse_phase(adapter,
497  tokens.pending_view(),
498  parents...);
499  if (keep_tokens &&
500  (match == parsing::pre_parse_action::valid_node)) {
501  match = parsing::pre_parse_action::skip_node_but_use_tokens;
502  }
503  }
504  });
505 
506  // If we have a result and it is false, then exit early. We need to
507  // wait until after name checking is (possibly) performed to throw an
508  // exception, otherwise a parse-stopping exception could be thrown when
509  // the target of the token isn't even this node
510  if (match == parsing::pre_parse_action::skip_node) {
511  return false;
512  }
513 
514  // Check the first token against the names (if the node is named of
515  // course)
516  if (is_named &&
517  (match != parsing::pre_parse_action::skip_node_but_use_tokens)) {
518  // If the node is named but there are no tokens, then take the first
519  // from args. If args is empty, then return false
520  if (result.empty()) {
521  if (tmp_args.empty()) {
522  return false;
523  }
524  result.push_back(tmp_args.front());
525  tmp_args.erase(tmp_args.begin());
526  }
527 
528  // The first token may not have been processed, so convert
529  auto& first_token = result.front();
530  if (first_token.prefix == parsing::prefix_type::none) {
531  first_token = parsing::get_token_type(first_token.name);
532  }
533 
534  // And then test it is correct unless we are skipping
535  if (!parsing::match<tree_node>(first_token)) {
536  return false;
537  }
538  }
539 
540  // If the policy checking returned an exception, now is the time to
541  // throw it
542  match.throw_exception();
543 
544  // Update args and tokens as we have a successful match
545  tokens.reserve(tokens.pending_view().size() + result.size());
546  tokens.insert_pending(tokens.pending_view().end(),
547  result.begin(),
548  result.end());
549  args = tmp_args;
550 
551  return match != parsing::pre_parse_action::skip_node;
552  }

605  {

610  // We have to wrap Fn because we're now giving it a

613  Fn,
