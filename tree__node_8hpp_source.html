164  template <typename Fn, typename ResultsTuple = std::tuple<>>
165  constexpr bool find(parsing::token_type token,
166  const Fn& visitor,
167  const ResultsTuple& results_tuple = {}) const
168  {
169  auto result = false;
170  utility::tuple_iterator(
171  [&](auto i, const auto& node) {
172  if (result) {
173  return;
174  }
175 
176  using node_type = std::decay_t<decltype(node)>;
177 
178  // Wrap the caller's visitor with one that forwards the child
179  // index
180  const auto wrapped_visitor = [&](const auto& child) {
181  visitor(i, child);
182  };
183 
184  constexpr auto match_arity = traits::arity_v<
185  decltype(&node_type::template match<
186  std::decay_t<decltype(wrapped_visitor)>>)>;
187 
188  if constexpr ((i < std::tuple_size_v<ResultsTuple>)&& //
189  (match_arity == 3)) {
190  if (node.match(token,
191  wrapped_visitor,
192  std::get<i.value>(results_tuple))) {
193  result = true;
194  }
195  } else {
196  if (node.match(token, wrapped_visitor)) {
197  result = true;
198  }
199  }
200  },
201  children());
202 
203  return result;
204  }
