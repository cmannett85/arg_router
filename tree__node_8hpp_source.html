408  template <typename Validator, bool HasTarget, typename Node, typename... Parents>
409  [[nodiscard]] std::optional<parsing::parse_target> pre_parse(
410  parsing::pre_parse_data<Validator, HasTarget> pre_parse_data,
411  const Node& node,
412  const Parents&... parents) const
413  {
414  auto result = vector<parsing::token_type>{};
415  auto tmp_args = pre_parse_data.args();
416  auto adapter = parsing::dynamic_token_adapter{result, tmp_args};
417 
418  // At this stage, the target is only for collecting sub-targets
419  auto target = parsing::parse_target{node, parents...};
420 
421  auto match = parsing::pre_parse_result{parsing::pre_parse_action::valid_node};
422  utility::tuple_type_iterator<priority_ordered_policies_type>([&](auto i) {
423  using policy_type = std::tuple_element_t<i, priority_ordered_policies_type>;
424  if constexpr (policy::has_pre_parse_phase_method_v<policy_type>) {
425  if (match == parsing::pre_parse_action::skip_node) {
426  return;
427  }
428 
429  // Keep the skip_node_but_use_sub_targets state if later policies return valid_node
430  const auto use_subs =
431  match == parsing::pre_parse_action::skip_node_but_use_sub_targets;
432  match = this->policy_type::pre_parse_phase(adapter,
433  extract_parent_target(pre_parse_data),
434  target,
435  node,
436  parents...);
437  if (use_subs && (match == parsing::pre_parse_action::valid_node)) {
438  match = parsing::pre_parse_action::skip_node_but_use_sub_targets;
439  }
440  }
441  });
442 
443  // If we have a result and it is false, then exit early. We need to wait until after name
444  // checking is (possibly) performed to throw an exception, otherwise a parse-stopping
445  // exception could be thrown when the target of the token isn't even this node
446  if (match == parsing::pre_parse_action::skip_node) {
447  return {};
448  }
449 
450  if constexpr (is_named) {
451  // Check the first token against the names (if the node is named of course)
452  if (match != parsing::pre_parse_action::skip_node_but_use_sub_targets) {
453  // If the node is named but there are no tokens, then take the first from args.
454  // If args is empty, then return false
455  if (result.empty()) {
456  if (tmp_args.empty()) {
457  return {};
458  }
459  result.push_back(tmp_args.front());
460  tmp_args.erase(tmp_args.begin());
461  }
462 
463  // The first token may not have been processed, so convert
464  auto& first_token = result.front();
465  if (first_token.prefix == parsing::prefix_type::none) {
466  first_token = parsing::get_token_type(first_token.name);
467  }
468 
469  // And then test it is correct unless we are skipping
470  if (!parsing::match<tree_node>(first_token)) {
471  return {};
472  }
473  }
474  }
475 
476  // If the policy checking returned an exception, now is the time to throw it
477  match.throw_exception();
478 
479  // If there are no processed tokens and no sub-targets then this node cannot do anything and
480  // therefore should not be used
481  if (result.empty() && target.sub_targets().empty()) {
482  return {};
483  }
484 
485  if (!pre_parse_data.validator()(node, parents...)) {
486  return {};
487  }
488 
489  // Update the unprocessed args
490  pre_parse_data.args() = tmp_args;
491 
492  // Update the target with the pre-parsed tokens. Remove the label token if present
493  if constexpr (is_named) {
494  if (!result.empty()) {
495  result.erase(result.begin());
496  }
497  }
498  target.tokens(std::move(result));
499 
500  return target;
501  }

544  template <template <typename...> typename Fn, typename... ExpandedParams>
545  [[nodiscard]] constexpr static auto common_filter(ExpandedParams&... expanded_params) noexcept
546  {
547  // Send references to the filter method so we don't copy anything unnecessarily
548  using ref_tuple = std::tuple<std::reference_wrapper<ExpandedParams>...>;
549 
550  // We have to wrap Fn because we're now giving it a std::reference_wrapper

552  boost::mp11::mp_bind<Fn, boost::mp11::mp_bind<traits::get_type, boost::mp11::_1>>;

557  ref_tuple{expanded_params...});

560  using result_t = boost::mp11::mp_transform<traits::get_type, ref_result_t>;
