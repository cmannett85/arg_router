350  template <typename Validator, bool HasTarget, typename Node, typename... Parents>
351  [[nodiscard]] std::optional<parsing::parse_target> pre_parse(
352  parsing::pre_parse_data<Validator, HasTarget> pre_parse_data,
353  const Node& node,
354  const Parents&... parents) const
355  {
356  auto result = vector<parsing::token_type>{};
357  auto tmp_args = pre_parse_data.args();
358  auto adapter = parsing::dynamic_token_adapter{result, tmp_args};
359 
360  // At this stage, the target is only for collecting sub-targets
361  auto target = parsing::parse_target{node, parents...};
362 
363  auto match = parsing::pre_parse_result{parsing::pre_parse_action::valid_node};
364  utility::tuple_type_iterator<priority_ordered_policies_type>([&](auto i) {
365  using policy_type = std::tuple_element_t<i, priority_ordered_policies_type>;
366  if constexpr (policy::has_pre_parse_phase_method_v<policy_type>) {
367  if (match == parsing::pre_parse_action::skip_node) {
368  return;
369  }
370 
371  // Keep the skip_node_but_use_sub_targets state if later policies return valid_node
372  const auto use_subs =
373  match == parsing::pre_parse_action::skip_node_but_use_sub_targets;
374  match = this->policy_type::pre_parse_phase(adapter,
375  extract_parent_target(pre_parse_data),
376  target,
377  node,
378  parents...);
379  if (use_subs && (match == parsing::pre_parse_action::valid_node)) {
380  match = parsing::pre_parse_action::skip_node_but_use_sub_targets;
381  }
382  }
383  });
384 
385  // If we have a result and it is false, then exit early. We need to wait until after name
386  // checking is (possibly) performed to throw an exception, otherwise a parse-stopping
387  // exception could be thrown when the target of the token isn't even this node
388  if (match == parsing::pre_parse_action::skip_node) {
389  return {};
390  }
391 
392  if constexpr (is_named) {
393  // Check the first token against the names (if the node is named of course)
394  if (match != parsing::pre_parse_action::skip_node_but_use_sub_targets) {
395  // If the node is named but there are no tokens, then take the first from args.
396  // If args is empty, then return false
397  if (result.empty()) {
398  if (tmp_args.empty()) {
399  return {};
400  }
401  result.push_back(tmp_args.front());
402  tmp_args.erase(tmp_args.begin());
403  }
404 
405  // The first token may not have been processed, so convert
406  auto& first_token = result.front();
407  if (first_token.prefix == parsing::prefix_type::none) {
408  first_token = parsing::get_token_type(first_token.name);
409  }
410 
411  // And then test it is correct unless we are skipping
412  if (!parsing::match<tree_node>(first_token)) {
413  return {};
414  }
415  }
416  }
417 
418  // If the policy checking returned an exception, now is the time to throw it
419  match.throw_exception();
420 
421  // If there are no processed tokens and no sub-targets then this node cannot do anything and
422  // therefore should not be used
423  if (result.empty() && target.sub_targets().empty()) {
424  return {};
425  }
426 
427  if (!pre_parse_data.validator()(node, parents...)) {
428  return {};
429  }
430 
431  // Update the unprocessed args
432  pre_parse_data.args() = tmp_args;
433 
434  // Update the target with the pre-parsed tokens. Remove the label token if present
435  if (is_named && !result.empty()) {
436  result.erase(result.begin());
437  }
438  target.tokens(std::move(result));
439 
440  return target;
441  }

485  [[nodiscard]] constexpr static auto common_filter(ExpandedParams&... expanded_params) noexcept
486  {
487  // Send references to the filter method so we don't copy anything unnecessarily
488  using ref_tuple = std::tuple<std::reference_wrapper<ExpandedParams>...>;
489 
490  // We have to wrap Fn because we're now giving it a std::reference_wrapper
491  using ref_fn =
492  boost::mp11::mp_bind<Fn, boost::mp11::mp_bind<traits::get_type, boost::mp11::_1>>;

497  ref_tuple{expanded_params...});

500  using result_t = boost::mp11::mp_transform<traits::get_type, ref_result_t>;
