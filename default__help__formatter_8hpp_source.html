34 template <std::size_t Indent>
35 class default_line_formatter
36 {
37  static_assert(Indent > 0, "Indent must be greater than zero");
38 
39 public:
49  template <std::size_t DescStart, std::size_t Depth, typename HelpData>
50  void format(std::ostream& stream, std::size_t columns)
51  {
52  if constexpr (!HelpData::label::empty()) {
53  constexpr auto indent = indent_size<Depth>();
54  stream << utility::create_sequence_cts_t<indent, ' '>::get() << HelpData::label::get();
55 
56  if constexpr (!HelpData::description::empty()) {
57  static_assert(HelpData::description::get().find('\t') == std::string_view::npos,
58  "Help descriptions cannot contain tabs");
59 
60  constexpr auto gap =
61  DescStart - indent - utility::utf8::terminal_width(HelpData::label::get());
62 
63  // Spacing between the end of the args label and start of description
64  stream << utility::create_sequence_cts_t<gap, ' '>::get();
65 
66  // Print the description, breaking if a word will exceed the terminal width
67  for (auto it = utility::utf8::line_iterator{HelpData::description::get(),
68  columns - DescStart};
69  it != utility::utf8::line_iterator{};) {
70  stream << *it;
71 
72  // If there's more data to follow, then add the offset
73  if (++it != utility::utf8::line_iterator{}) {
74  stream << '\n' << utility::create_sequence_cts_t<DescStart, ' '>::get();
75  }
76  }
77  }
78 
79  stream << "\n";
80  }
81  }
82 
83 private:
84  template <std::size_t Depth>
85  [[nodiscard]] constexpr static std::size_t indent_size() noexcept
86  {
87  return Depth * Indent;
88  }
89 };

174 template <typename Indent = traits::integral_constant<std::size_t{4}>,
175  typename DescColumnOffset = traits::integral_constant<Indent{} * 2>,
176  typename LineFormatter = help_formatter_component::default_line_formatter<Indent{}>,
177  typename PreambleFormatter = help_formatter_component::default_preamble_formatter,
178  typename AddendumFormatter = help_formatter_component::default_addendum_formatter>
179 class default_help_formatter_t
180 {
181  static_assert(traits::has_value_type_v<Indent>, "Indent must have a value_type");
182  static_assert(Indent{} > 0, "Indent value_type must be greater than zero");
183  static_assert(traits::has_value_type_v<DescColumnOffset>,
184  "DescColumnOffset must have a value_type");
185  static_assert(DescColumnOffset{} > 0, "DescColumnOffset value_type must be greater than zero");
186 
187 public:
199  template <typename Node, typename HelpNode, bool Flatten>
200  static void generate_help(std::ostream& stream)
201  {
202  static_assert(traits::has_help_data_type_v<Node>,
203  "Node must have a help_data_type to generate help from");
204 
205  using help_data_type = typename Node::template help_data_type<Flatten>;
206 
207  // Write out the preamble
208  auto preamble_formatter = PreambleFormatter{};
209  preamble_formatter.template format<HelpNode>(stream);
210 
211  // Calculate description offset
212  constexpr auto desc_column = description_column_start<0, help_data_type>(0);
213 
214  // Get the current number of console columns, so we can wrap the description nicely. If
215  // the call fails (it returns zero), or the description field start column plus a fixed
216  // offset exceeds the column count, then don't attempt to wrap
217  const auto columns = utility::terminal::columns();
218 
219  // Generate the args output
220  auto line_formatter = LineFormatter{};
221  generate_help_impl<desc_column, 0, help_data_type>(
222  stream,
223  columns >= (desc_column + DescColumnOffset{}) ? columns :
224  std::numeric_limits<std::size_t>::max(),
225  line_formatter);
226 
227  // Write out the addendum
228  auto addendum_formatter = AddendumFormatter{};
229  addendum_formatter.template format<HelpNode>(stream);
230  }
231 
232 private:
233  template <std::size_t Depth, typename HelpData>
234  [[nodiscard]] constexpr static std::size_t description_column_start(
235  std::size_t current_max) noexcept
236  {
237  constexpr auto this_row_start =
238  (Depth * Indent{}) + utility::utf8::terminal_width(HelpData::label::get()) + Indent{};
239  current_max = std::max(current_max, this_row_start);
240 
241  utility::tuple_type_iterator<typename HelpData::children>([&](auto i) {
242  using child_type = std::tuple_element_t<i, typename HelpData::children>;
243 
244  current_max = description_column_start<Depth + 1, child_type>(current_max);
245  });
246 
247  return current_max;
248  }
249 
250  template <std::size_t DescStart, std::size_t Depth, typename HelpData>
251  static void generate_help_impl(std::ostream& stream,
252  std::size_t columns,
253  LineFormatter& line_formatter)
254  {
255  line_formatter.template format<DescStart, Depth, HelpData>(stream, columns);
256 
257  utility::tuple_type_iterator<typename HelpData::children>([&](auto i) {
258  using child_type = std::tuple_element_t<i, typename HelpData::children>;
259 
260  generate_help_impl<DescStart, Depth + 1, child_type>(stream, columns, line_formatter);
261  });
262  }
263 };
