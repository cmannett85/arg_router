300 template <typename ChildrenTuple, typename Fn>
301 bool visit_child(const token_type& token,
302  const ChildrenTuple& children,
303  Fn&& visitor)
304 {
305  // Iterate over the children and use the prefix type to use the correct
306  // name method
307  auto found_child = false;
308  utility::tuple_iterator(
309  [&](auto i, auto&& child) {
310  using child_type = std::decay_t<decltype(child)>;
311 
312  if constexpr (traits::is_detected_v<parsing::has_match_checker,
313  child_type>) {
314  const auto result = child.match(token);
315  if (result.matched) {
316  visitor(i, child, result);
317  found_child = true;
318  }
319  } else if (token.prefix == prefix_type::NONE) {
320  // A positional arg type will always accept a token
321  visitor(
322  i,
323  child,
324  parsing::match_result{parsing::match_result::NO_MATCH,
325  parsing::match_result::HAS_ARGUMENT});
326  found_child = true;
327  }
328 
329  // No need to check that it has matched again, as the compiler
330  // has already done that via a rule
331  },
332  children);
333 
334  return found_child;
335 }
