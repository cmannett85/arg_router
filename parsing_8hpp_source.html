159 template <typename ChildrenTuple, typename Fn>
160 bool visit_child(const token_type& token,
161  const ChildrenTuple& children,
162  Fn visitor)
163 {
164  auto found_child = false;
165  utility::tuple_iterator(
166  [&](auto i, auto&& child) {
167  using child_type = std::decay_t<decltype(child)>;
168 
169  auto call_visitor = [&]() {
170  visitor(i, child);
171  found_child = true;
172  };
173 
174  if constexpr (traits::has_match_method_v<child_type>) {
175  const auto result = child.match(token);
176  if (result) {
177  // Skip if already found
178  if (found_child) {
179  return;
180  }
181 
182  call_visitor();
183  }
184  } else if (node_category::is_positional_arg_like_v<child_type>) {
185  // A positional arg type will always accept a token
186  call_visitor();
187  }
188  },
189  children);
190 
191  return found_child;
192 }

215 template <typename ChildrenTuple, typename RouterArgs, typename Fn>
216 bool visit_child(const token_type& token,
217  const ChildrenTuple& children,
218  const RouterArgs& router_args,
219  Fn visitor)
220 {
221  static_assert(
222  std::tuple_size_v<ChildrenTuple> == std::tuple_size_v<RouterArgs>,
223  "Number of children must be the same as number of router args");
224 
225  auto found_child = false;
226  auto wrapped_visitor = [&](auto i, const auto& child) {
227  using child_type = std::tuple_element_t<i, ChildrenTuple>;
228  using value_type =
229  typename std::tuple_element_t<i, RouterArgs>::value_type;
230 
231  // Skip if we have already found
232  if (found_child) {
233  return;
234  }
235 
236  // Non-positional arg case just forwards onto the original visitor
237  if constexpr (traits::has_match_method_v<child_type>) {
238  visitor(i, child);
239  found_child = true;
240  } else {
241  if constexpr (traits::has_push_back_method_v<value_type>) {
242  // Unpleasant, there's no 'if constexpr' ternary operator though
243  constexpr auto max_count = [&]() {
244  if constexpr (traits::has_maximum_count_method_v<
245  child_type>) {
246  return child_type::maximum_count();
247  }
248 
249  return std::numeric_limits<std::size_t>::max();
250  }();
251 
252  // Compare the number of values the position arg has already, to
253  // check if its maximum has been reached. If the maximum has
254  // been reached then just skip onto the next
255  const auto& value = std::get<i>(router_args);
256  const auto num_values = value ? std::size(*value) : 0u;
257  if (num_values < max_count) {
258  visitor(i, child);
259  found_child = true;
260  }
261  } else {
262  // Positional args support non-container value_types if their
263  // min/max/count values are the same. So use the hit_mask to
264  // determine whether or not to skip
265  if (!std::get<i>(router_args)) {
266  visitor(i, child);
267  found_child = true;
268  }
269  }
270  }
271  };
272 
273  visit_child(token, children, wrapped_visitor);
274  return found_child;
275 }

280 void find_target_mode(const Node& node, token_list& tokens, Fn visitor)
281 {
282  // If there's no prefix then it might be a named mode-like type, so find
283  // the matching child.
284  const auto& token = tokens.front();
285  auto found_mode = false;
286  if (token.prefix == parsing::prefix_type::NONE) {
287  utility::tuple_iterator(
288  [&](auto /*i*/, auto&& child) {
289  using child_type = std::decay_t<decltype(child)>;
290 
291  if constexpr (node_category::is_named_mode_like_v<child_type>) {
292  if (child.match(token)) {
293  found_mode = true;
294  tokens.pop_front();
295  find_target_mode(child, tokens, std::move(visitor));
296  }
297  }
298  },
299  node.children());
300  }
301 
302  // If there's no named mode, then we have our target (assuming the request
303  // was valid)
304  if (!found_mode) {
305  visitor(node, tokens);
306  }

334 template <typename Node, typename Fn>
335 void find_target_node(const Node& node, token_list& tokens, Fn visitor)
336 {
337  using children_type = typename Node::children_type;
338 
339  // If there are no tokens, find the anonymous mode-like type
340  if (tokens.empty()) {
341  constexpr auto anonymous_mode_index = boost::mp11::mp_find_if<
342  children_type,
343  node_category::is_anonymous_mode_like>::value;
344 
345  if constexpr (anonymous_mode_index ==
346  std::tuple_size_v<children_type>) {
347  throw parse_exception{"No arguments provided"};
348  } else {
349  visitor(std::get<anonymous_mode_index>(node.children()), tokens);
350  return;
351  }
352  }
353 
354  const auto found_child = parsing::visit_child(
355  tokens.front(),
356  node.children(),
357  [&](auto /*i*/, const auto& child) {
358  using child_type = std::decay_t<decltype(child)>;
359 
360  if (node_category::is_named_mode_like_v<child_type>) {
361  tokens.pop_front();
362  detail::find_target_mode(child, tokens, std::move(visitor));
363  } else {
364  visitor(child, tokens);
365  }
366  });
367 
368  if (!found_child) {
369  throw parse_exception{"Unknown argument", tokens.front()};
370  }
