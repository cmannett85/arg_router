307 template <typename ChildrenTuple, typename Fn>
308 bool visit_child(const token_type& token,
309  const ChildrenTuple& children,
310  Fn visitor)
311 {
312  auto found_child = false;
313  utility::tuple_iterator(
314  [&](auto i, auto&& child) {
315  using child_type = std::decay_t<decltype(child)>;
316 
317  auto call_visitor = [&]() {
318  visitor(i, child);
319  found_child = true;
320  };
321 
322  if constexpr (traits::is_detected_v<parsing::has_match_checker,
323  child_type>) {
324  const auto result = child.match(token);
325  if (result) {
326  // Skip if already found
327  if (found_child) {
328  return;
329  }
330 
331  call_visitor();
332  }
333  } else if (token.prefix == prefix_type::NONE &&
334  policy::has_value_tokens_v<child_type>) {
335  // A positional arg type will always accept a token
336  call_visitor();
337  }
338  },
339  children);

365 template <typename ChildrenTuple, typename RouterArgs, typename Fn>
366 bool visit_child(const token_type& token,
367  const ChildrenTuple& children,
368  const RouterArgs& router_args,
369  Fn visitor)
370 {
371  static_assert(
372  std::tuple_size_v<ChildrenTuple> == std::tuple_size_v<RouterArgs>,
373  "Number of children must be the same as number of router args");
374 
375  auto found_child = false;
376  auto wrapped_visitor = [&](auto i, const auto& child) {
377  using child_type = std::tuple_element_t<i, ChildrenTuple>;
378  using value_type =
379  typename std::tuple_element_t<i, RouterArgs>::value_type;
380 
381  // Skip if we have already found
382  if (found_child) {
383  return;
384  }
385 
386  // Non-positional arg case just forwards onto the original visitor
387  if constexpr (traits::is_detected_v<parsing::has_match_checker,
388  child_type>) {
389  visitor(i, child);
390  found_child = true;
391  } else {
392  if constexpr (traits::is_detected_v<has_push_back_checker,
393  value_type>) {
394  // Unpleasant, there's no 'if constexpr' ternary operator though
395  constexpr auto max_count = [&]() {
396  if constexpr (traits::is_detected_v<
397  has_maximum_count_checker,
398  child_type>) {
399  return child_type::maximum_count();
400  }
401 
402  return std::numeric_limits<std::size_t>::max();
403  }();
404 
405  // Compare the number of values the position arg has already, to
406  // check if its maximum has been reached. If the maximum has
407  // been reached then just skip onto the next
408  const auto& value = std::get<i>(router_args);
409  const auto num_values = value ? std::size(*value) : 0u;
410  if (num_values < max_count) {
411  visitor(i, child);
412  found_child = true;
413  }
414  } else {
415  // Positional args support non-container value_types if their
416  // min/max/count values are the same. So use the hit_mask to
417  // determine whether or not to skip
418  if (!std::get<i>(router_args)) {
419  visitor(i, child);
420  found_child = true;
421  }
422  }
423  }
424  };
425 
426  visit_child(token, children, wrapped_visitor);
427  return found_child;
428 }
