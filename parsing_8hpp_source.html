36 template <typename Node, typename Fn, std::size_t N>
37 [[nodiscard]] constexpr bool find(const Node& node,
38  parsing::token_type token,
39  std::bitset<N> hit_flags,
40  const Fn& visitor)
41 {
42  auto result = false;
43  utility::tuple_iterator(
44  [&](auto i, const auto& node) {
45  if (result || hit_flags[i]) {
46  return;
47  }
48 
49  // Wrap the caller's visitor with one that forwards the child
50  // index
51  const auto wrapped_visitor = [&](const auto& child) {
52  visitor(i, child);
53  };
54 
55  if (node.match(token, wrapped_visitor)) {
56  result = true;
57  }
58  },
59  node.children());
60 
61  return result;

64 template <typename Node>
65 void expand_arguments(const Node& node,
66  span<const char*>& args,
67  token_list& result)
68 {
69  using namespace std::string_literals;
70  using namespace utility::string_view_ops;
71 
72  using hit_flags_type =
73  std::bitset<std::tuple_size_v<typename Node::children_type>>;
74 
75  auto hit_flags = hit_flags_type{};
76  while (!args.empty()) {
77  // Convert to a token_type and try to find the matching node
78  const auto tt = get_token_type(args.front());
79 
80  // Expand out if it's collapsed short-form flags, this is valid because
81  // no value tokens are at the start of token processing chain - unless
82  // this is for positional arguments but that's user ambiguity which we
83  // can't eliminate (it's not psychic)
84  if (tt.prefix == prefix_type::SHORT && tt.name.size() > 1u) {
85  // Expand it out
86  for (auto i = 0u; i < tt.name.size(); ++i) {
87  result.emplace_pending(prefix_type::SHORT,
88  std::string_view{&(tt.name[i]), 1});
89  }
90  args = args.subspan(1);
91  continue;
92  }
93 
94  const auto consume = [&]() {
95  result.add_pending(tt);
96  args = args.subspan(1);
97  };
98 
99  const auto found_node =
100  find(node, tt, hit_flags, [&](auto i, const auto& child) {
101  // We have a match, but is it just a positional_arg absorbing
102  // everything? Check by looking at it's name policies
103  using child_type = std::decay_t<decltype(child)>;
104 
105  constexpr auto is_named =
106  traits::has_long_name_method_v<child_type> ||
107  traits::has_short_name_method_v<child_type> ||
108  traits::has_none_name_method_v<child_type>;
109 
110  // Don't revisit nodes, this prevents the first positional_arg
111  // from consuming everything
112  hit_flags[i] = !policy::has_multi_stage_value_v<child_type>;
113 
114  if (tt.prefix == prefix_type::LONG) {
115  if constexpr (traits::has_value_separator_method_v<
116  child_type>) {
117  const auto split_token =
118  detail::handle_value_separator<child_type>(tt);
119  if (!split_token) {
120  throw parse_exception{
121  "Expected to find value separator '"s +
122  child_type::value_separator() + "' in \"" +
123  args[0] + "\""};
124  }
125  result.add_pending(*split_token);
126 
127  // Update the args entry to point just to the value part
128  const auto offset =
129  split_token->name.size() +
130  child_type::value_separator().size();
131  args[0] +=
132  to_string(split_token->prefix).size() + offset;
133 
134  // Someone has put the value separator but then no value
135  // after it...
136  if (offset == tt.name.size()) {
137  throw parse_exception{
138  "Unable to find value after separator",
139  tt};
140  }
141  } else if constexpr (is_named) {
142  // It's not a positional_arg, so add the name token
143  consume();
144  }
145  } else if (is_named) {
146  // It's not a positional_arg, so add the name token
147  consume();
148  }
149 
150  // Using the count of the node, add the correct number of NONE
151  // (i.e. value) tokens to the list
152  if constexpr (traits::has_maximum_count_method_v<child_type>) {
153  const auto value_count =
154  std::min(child_type::maximum_count(), args.size());
155  if (value_count > 0u) {
156  result.reserve(result.pending_view().size() +
157  value_count);
158  for (auto i = 0u; i < value_count; ++i) {
159  result.emplace_pending(prefix_type::NONE, args[i]);
160  }
161  args = args.subspan(value_count);
162  }
163  } else if constexpr (traits::has_process_value_tokens_method_v<
164  child_type>) {
165  child_type::process_value_tokens(args, result);
166  }
167 
168  // Recurse using the found node if necessary
169  if constexpr ((std::tuple_size_v<
170  typename child_type::children_type>) > 0) {
171  if (!args.empty()) {
172  expand_arguments(child, args, result);
173  }
174  }
175  });
176 
177  // If we can't find the node, just add it as our best guess. We do not
178  // error here as the purpose of this parsing phase is to categorise as
179  // best we can the tokens, better error checking happens later
180  if (!found_node) {
181  consume();
182  }
183  }
184 }

218 template <typename T>
219 [[nodiscard]] constexpr bool default_match(token_type token) noexcept
220 {
221  using namespace std::string_literals;
222 
223  if constexpr (traits::has_long_name_method_v<T>) {
224  if (token.prefix == prefix_type::LONG) {
225  if constexpr (traits::has_value_separator_method_v<T>) {
226  const auto split_token =
227  detail::handle_value_separator<T>(token);
228  if (split_token) {
229  token = *split_token;
230  }
231  }
232 
233  if (token.name == T::long_name()) {
234  return true;
235  }
236  }
237  }
238  if constexpr (traits::has_short_name_method_v<T>) {
239  if ((token.prefix == prefix_type::SHORT) &&
240  (token.name == T::short_name())) {
241  return true;
242  }
243  }
244  if constexpr (traits::has_none_name_method_v<T>) {
245  if ((token.prefix == prefix_type::NONE) &&
246  (token.name == T::none_name())) {
247  return true;
248  }
249  }
250 
251  return false;
252 }

262 template <typename Node>
