157 template <typename ChildrenTuple, typename Fn>
158 bool visit_child(const token_type& token,
159  const ChildrenTuple& children,
160  Fn visitor)
161 {
162  auto found_child = false;
163  utility::tuple_iterator(
164  [&](auto i, auto&& child) {
165  using child_type = std::decay_t<decltype(child)>;
166 
167  auto call_visitor = [&]() {
168  visitor(i, child);
169  found_child = true;
170  };
171 
172  if constexpr (traits::has_match_method_v<child_type>) {
173  const auto result = child.match(token);
174  if (result) {
175  // Skip if already found
176  if (found_child) {
177  return;
178  }
179 
180  call_visitor();
181  }
182  } else if (token.prefix == prefix_type::NONE &&
183  policy::has_value_tokens_v<child_type>) {
184  // A positional arg type will always accept a token
185  call_visitor();
186  }
187  },
188  children);
189

191 }

214 template <typename ChildrenTuple, typename RouterArgs, typename Fn>
215 bool visit_child(const token_type& token,
216  const ChildrenTuple& children,
217  const RouterArgs& router_args,
218  Fn visitor)
219 {
220  static_assert(
221  std::tuple_size_v<ChildrenTuple> == std::tuple_size_v<RouterArgs>,
222  "Number of children must be the same as number of router args");
223 
224  auto found_child = false;
225  auto wrapped_visitor = [&](auto i, const auto& child) {
226  using child_type = std::tuple_element_t<i, ChildrenTuple>;
227  using value_type =
228  typename std::tuple_element_t<i, RouterArgs>::value_type;
229 
230  // Skip if we have already found
231  if (found_child) {
232  return;
233  }
234 
235  // Non-positional arg case just forwards onto the original visitor
236  if constexpr (traits::has_match_method_v<child_type>) {
237  visitor(i, child);
238  found_child = true;
239  } else {
240  if constexpr (traits::has_push_back_method_v<value_type>) {
241  // Unpleasant, there's no 'if constexpr' ternary operator though
242  constexpr auto max_count = [&]() {
243  if constexpr (traits::has_maximum_count_method_v<
244  child_type>) {
245  return child_type::maximum_count();
246  }
247 
248  return std::numeric_limits<std::size_t>::max();
249  }();
250 
251  // Compare the number of values the position arg has already, to
252  // check if its maximum has been reached. If the maximum has
253  // been reached then just skip onto the next
254  const auto& value = std::get<i>(router_args);
255  const auto num_values = value ? std::size(*value) : 0u;
256  if (num_values < max_count) {
257  visitor(i, child);
258  found_child = true;
259  }
260  } else {
261  // Positional args support non-container value_types if their
262  // min/max/count values are the same. So use the hit_mask to
263  // determine whether or not to skip
264  if (!std::get<i>(router_args)) {
265  visitor(i, child);
266  found_child = true;
267  }
268  }
269  }
270  };
271 
272  visit_child(token, children, wrapped_visitor);
273  return found_child;
274 }
