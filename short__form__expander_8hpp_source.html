48  template <typename ProcessedTarget, typename... Parents>
49  [[nodiscard]] parsing::pre_parse_result pre_parse_phase(
50  parsing::dynamic_token_adapter& tokens,
51  [[maybe_unused]] utility::compile_time_optional<ProcessedTarget> processed_target,
52  [[maybe_unused]] parsing::parse_target& target,
53  [[maybe_unused]] const Parents&... parents) const
54  {
55  static_assert(sizeof...(Parents) > 0,
56  "Short-form expansion policy requires at least one parent");
57  using owner_type = boost::mp11::mp_first<std::tuple<Parents...>>;
58 
59  static_assert(traits::has_short_name_method_v<owner_type>,
60  "Short-form expansion support requires a short name policy");
61 
62  static_assert(utility::utf8::count(owner_type::short_name()) == 1,
63  "Short name must only be 1 character");
64 
65  // Parents is always greater than zero, but having it in this statement causes it to only be
66  // evaluated upon template instantiation - otherwise it would fail even if the class is
67  // never used
68  static_assert((sizeof...(Parents) > 0) && (config::short_prefix != config::long_prefix),
69  "Short and long prefixes cannot be the same");
70 
71  if (tokens.empty()) {
72  return parsing::pre_parse_action::valid_node;
73  }
74 
75  auto first = tokens.begin();
76  auto first_token = *first;
77  if (first_token.prefix == parsing::prefix_type::none) {
78  // The token has _probably_ not been processed yet, so try to convert to short form
79  const auto tt = parsing::get_token_type(first_token.name);
80  if (tt.prefix != parsing::prefix_type::short_) {
81  return parsing::pre_parse_action::valid_node;
82  }
83 
84  first_token = tt;
85  } else if (first_token.prefix == parsing::prefix_type::long_) {
86  return parsing::pre_parse_action::valid_node;
87  }
88 
89  // Exit early if there's no expansion to be done
90  if (first_token.name.size() == owner_type::short_name().size()) {
91  return parsing::pre_parse_action::valid_node;
92  }
93 
94  // Move the token to the processed container
95  tokens.transfer(first);
96 
97  // Insert the extra flags at the front of the unprocessed section, so they will be processed
98  // independently
99  tokens.unprocessed().reserve(tokens.unprocessed().size() + first_token.name.size() - 1);
100  auto it = tokens.unprocessed().begin();
101 
102  // Skip past the first grapheme cluster as we'll re-use the existing short form token for
103  // that
104  for (auto gc_it = ++utility::utf8::iterator{first_token.name};
105  gc_it != utility::utf8::iterator{};
106  ++gc_it, ++it) {
107  it = tokens.unprocessed().insert(it, {parsing::prefix_type::short_, *gc_it});
108  }
109 
110  // Shrink the first to a single grapheme cluster
111  first.set({parsing::prefix_type::short_, *utility::utf8::iterator{first_token.name}});
112 
113  return parsing::pre_parse_action::valid_node;
114  }
115 };
