123 template <typename... Policies>
124 [[nodiscard]] constexpr auto flag(Policies... policies) noexcept
125 {
126  // Add the short-form expander if one of the policies implements the short name method, and if
127  // the short and long prefix are not the same
128  constexpr auto has_short_name = boost::mp11::mp_any_of<std::tuple<std::decay_t<Policies>...>,
129  traits::has_short_name_method>::value;
130  if constexpr ((config::long_prefix != config::short_prefix) && has_short_name) {
131  return flag_t{policy::short_form_expander, std::move(policies)...};
