131 template <typename T, typename... Policies>
132 [[nodiscard]] constexpr auto counting_flag(Policies... policies) noexcept
133 {
134  // Add the short-form expander if one of the policies implements the short name method, and if
135  // the short and long prefix are not the same
136  constexpr auto has_short_name = boost::mp11::mp_any_of<std::tuple<std::decay_t<Policies>...>,
137  traits::has_short_name_method>::value;
138  if constexpr ((config::long_prefix != config::short_prefix) && has_short_name) {
139  return counting_flag_t<T, policy::short_form_expander_t<>, std::decay_t<Policies>...>{
140  policy::short_form_expander,
141  std::move(policies)...};
142  } else {
143  return counting_flag_t<T, std::decay_t<Policies>...>{std::move(policies)...};
144  }
