197  template <typename Node, typename TargetFn>
198  static void find_help_target(vector<parsing::token_type>& tokens,
199  const Node& node,
200  const TargetFn& fn)
201  {
202  if (tokens.empty()) {
203  fn(node);
204  return;
205  }
206 
207  // Help tokens aren't pre-parsed by the target nodes (as they would fail if missing any
208  // required value tokens), so we have just use the prefix to generate a token_type from
209  // them, as they all of a prefix_type of none
210  const auto token = parsing::get_token_type(tokens.front().name);
211 
212  auto result = false;
213  utility::tuple_iterator(
214  [&](auto /*i*/, const auto& child) {
215  using child_type = std::decay_t<decltype(child)>;
216 
217  if (!result && parsing::match<child_type>(token)) {
218  result = true;
219  tokens.erase(tokens.begin());
220  find_help_target(tokens, child, fn);
221  }
222  },
223  node.children());
224 
225  if (!result) {
226  throw parse_exception{"Unknown argument", token};
227  }
228  }

254  template <std::size_t DescStart, std::size_t Depth, typename HelpData>
255  static void generate_help(std::ostream& stream, std::size_t columns)
256  {
257  if constexpr (!HelpData::label::empty()) {
258  constexpr auto indent = indent_size<Depth>();
259  stream << utility::create_sequence_cts_t<indent, ' '>::get() << HelpData::label::get();
260 
261  if constexpr (!HelpData::description::empty()) {
262  static_assert(HelpData::description::get().find('\t') == std::string_view::npos,
263  "Help descriptions cannot contain tabs");
264 
265  constexpr auto gap =
266  DescStart - indent - utility::utf8::terminal_width(HelpData::label::get());
267 
268  // Spacing between the end of the args label and start of description
269  stream << utility::create_sequence_cts_t<gap, ' '>::get();
270 
271  // Print the description, breaking if a word will exceed the terminal width
272  for (auto it = utility::utf8::line_iterator{HelpData::description::get(),
273  columns - DescStart};
274  it != utility::utf8::line_iterator{};) {
275  stream << *it;
276 
277  // If there's more data to follow, then add the offset
278  if (++it != utility::utf8::line_iterator{}) {
279  stream << '\n' << utility::create_sequence_cts_t<DescStart, ' '>::get();
280  }
281  }
282  }
283 
284  stream << "\n";
285  }
286 
287  utility::tuple_type_iterator<typename HelpData::children>([&](auto i) {
288  using child_type = std::tuple_element_t<i, typename HelpData::children>;
289 
290  generate_help<DescStart, Depth + 1, child_type>(stream, columns);
291  });
