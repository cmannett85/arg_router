142  template <typename... Parents>
143  void parse(parsing::parse_target&& target, const Parents&... parents) const
144  {
145  const auto& root = std::get<sizeof...(Parents) - 1>( //
146  std::tuple{std::cref(parents)...})
147  .get();
148 
149  find_help_target(target.tokens(), root, [&](const auto& target_node) {
150  using root_type = std::decay_t<decltype(root)>;
151  using node_type = std::decay_t<decltype(target_node)>;
152 
153  // If the user has specified a help target then enable flattening otherwise the output
154  // is a bit useless...
155  constexpr auto flatten =
156  algorithm::has_specialisation_v<policy::flatten_help_t,
157  typename parent_type::policies_type> ||
158  !std::is_same_v<root_type, node_type>;
159 
160  // If there is a routing policy then delegate to it, otherwise print the help output to
161  // std::cout and exit
162  using routing_policy =
163  typename parent_type::template phase_finder_t<policy::has_routing_phase_method>;
164  if constexpr (std::is_void_v<routing_policy>) {
165  formatter_type::template generate_help<node_type, help_t, flatten>(std::cout);
166  std::exit(EXIT_SUCCESS);
167  } else {
168  auto stream = ostringstream{};
169  formatter_type::template generate_help<node_type, help_t, flatten>(stream);
170  this->routing_policy::routing_phase(std::move(stream));
171  }
172  });
173  }
174 
175 private:
176  static_assert(
177  !parent_type::template any_phases_v<bool, // Type doesn't matter, as long as it isn't void
178  policy::has_parse_phase_method,
179  policy::has_validation_phase_method,
180  policy::has_missing_phase_method>,
181  "Help only supports policies with pre-parse and routing phases");
182 
183  template <typename Node, typename TargetFn>
184  static void find_help_target(vector<parsing::token_type>& tokens,
185  const Node& node,
186  const TargetFn& fn)
187  {
188  if (tokens.empty()) {
189  fn(node);
190  return;
191  }
192 
193  // Help tokens aren't pre-parsed by the target nodes (as they would fail if missing any
194  // required value tokens), so we have just use the prefix to generate a token_type from
195  // them, as they all of a prefix_type of none
196  const auto token = parsing::get_token_type(tokens.front().name);
197 
198  auto result = false;
199  utility::tuple_iterator(
200  [&](auto /*i*/, const auto& child) {
201  using child_type = std::decay_t<decltype(child)>;
202 
203  if (!result && parsing::match<child_type>(token)) {
204  result = true;
205  tokens.erase(tokens.begin());
206  find_help_target(tokens, child, fn);
207  }
208  },
209  node.children());
210 
211  if (!result) {
212  throw multi_lang_exception{error_code::unknown_argument, token};
213  }
214  }
