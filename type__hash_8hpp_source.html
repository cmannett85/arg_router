11 // constexpr hash combine implementations, based on Boost's equivalent:
12 // https://github.com/boostorg/container_hash/blob/boost-1.74.0/include/boost/container_hash/hash.hpp#L316
13 [[nodiscard]] constexpr std::size_t hash_combine(std::size_t a, std::size_t b) noexcept
14 {
15  static_assert((sizeof(std::size_t) == 4) || (sizeof(std::size_t) == 8),
16  "std::size_t must be 32 or 64 bits");
17 
18  // Personally I hate this and previously had it as two separate overloads, but that caused
19  // ambiguity on AppleClang hence approach
20  if constexpr (sizeof(std::size_t) == 4) {
21  constexpr auto c1 = std::uint32_t{0xcc9e2d51};
22  constexpr auto c2 = std::uint32_t{0x1b873593};
23 
24  // NOLINTBEGIN(readability-magic-numbers)
25  b *= c1;
26  b = (b << 15) | (b >> (32 - 15)); // ROTL 15 bits
27  b *= c2;
28 
29  a ^= b;
30  a = (a << 13) | (a >> (32 - 13)); // ROTL 13 bits
31  a = a * 5 + 0xe6546b64;
32  // NOLINTEND(readability-magic-numbers)
33 
34  return a;
35  } else {
36  constexpr auto m = std::uint64_t{0xc6a4a7935bd1e995};
37  constexpr auto r = 47;
38 
39  b *= m;
40  b ^= b >> r;
41  b *= m;
42 
43  a ^= b;
44  a *= m;
45 
46  // Completely arbitrary number, to prevent 0's from hashing to 0.
47  // NOLINTNEXTLINE(readability-magic-numbers)
48  a += 0xe6546b64;
49 
50  return a;
51  }
52 }

54 // Do this rather than just put the generate() body into type_hash(), because otherwise the compiler
55 // puts the pretty function output into static storage, needlessly bloating exe size
56 template <typename T>
57 class type_hash_t
58 {
59  [[nodiscard]] static constexpr std::size_t generate() noexcept
60  {
61  // Because we can't guarantee default construction support for T and reinterpret_cast is not
62  // allowed in constant evaluation, we'll resort to the (valid, but) dirty hack of using the
63  // function signature (which contains the fully qualified name of T)
64 #ifdef _MSC_VER
65  constexpr auto sig = std::string_view{__FUNCSIG__};
66 #else
67  constexpr auto sig = std::string_view{static_cast<const char*>(__PRETTY_FUNCTION__)};
68 #endif
69 
70  auto result = std::size_t{0};
71  for (auto c : sig) {
72  result = detail::hash_combine(result, c);
73  }
74 
75  return result;
76  }
77 
78 public:
79  static constexpr auto value = generate();
80 };
81 } // namespace detail
