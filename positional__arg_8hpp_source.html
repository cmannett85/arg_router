130  value_type parse(parsing::token_list& tokens,
131  const Parents&... parents) const
132  {
133  // Pre-parse
134  utility::tuple_type_iterator<policies_type>([&](auto i) {
135  using policy_type = std::tuple_element_t<i, policies_type>;
136  if constexpr (policy::has_pre_parse_phase_method_v<policy_type>) {
137  this->policy_type::pre_parse_phase(tokens, *this, parents...);
138  }
139  });
140 
141  auto view = tokens.pending_view();
142 
143  auto result = value_type{};
144  if constexpr (traits::has_push_back_method_v<value_type>) {
145  if constexpr (traits::has_maximum_count_method_v<
146  positional_arg_t>) {
147  view = view.subspan(0, positional_arg_t::maximum_count());
148  }
149 
150  for (auto token : view) {
151  result.push_back(
152  parent_type::template parse<value_type>(token.name,
153  *this,
154  parents...));
155  }
156  } else if (!view.empty()) {
157  result = parent_type::template parse<value_type>(view.front().name,
158  *this,
159  parents...);
160  }
161 
162  // Pop the tokens, we don't need them anymore
163  tokens.mark_as_processed(view.size());
164 
165  // Validation
166  utility::tuple_type_iterator<policies_type>([&](auto i) {
167  using policy_type = std::tuple_element_t<i, policies_type>;
168  if constexpr (policy::has_validation_phase_method_v<policy_type,
169  value_type>) {
170  this->policy_type::validation_phase(result, *this, parents...);
171  }
172  });
173 
174  // No routing phase, a positional_arg cannot be used as a top-level node
175 
176  return result;
177  }
