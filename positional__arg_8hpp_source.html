145  template <typename... Parents>
146  [[nodiscard]] value_type parse(parsing::parse_target target, const Parents&... parents) const
147  {
148  auto result = value_type{};
149  if constexpr (traits::has_push_back_method_v<value_type>) {
150  for (auto token : target.tokens()) {
151  result.push_back(
152  parent_type::template parse<value_type>(token.name, *this, parents...));
153  }
154  } else if (!target.tokens().empty()) {
155  result = parent_type::template parse<value_type>(target.tokens().front().name,
156  *this,
157  parents...);
158  }
159 
160  // Validation
161  utility::tuple_type_iterator<policies_type>([&](auto i) {
162  using policy_type = std::tuple_element_t<i, policies_type>;
163  if constexpr (policy::has_validation_phase_method_v<policy_type, value_type>) {
164  this->policy_type::validation_phase(result, *this, parents...);
165  }
166  });
167 
168  // No routing phase, a positional_arg cannot be used as a top-level node
169 
170  return result;
171  }
172 
173 private:
174  static_assert(!parent_type::template any_phases_v<value_type, policy::has_routing_phase_method>,
175  "Positional arg does not support policies with routing phases "
176  "(e.g. router)");
177 };
