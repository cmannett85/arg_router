208  template <typename... Parents>
209  value_type parse(parsing::token_list& tokens,
210  const Parents&... parents) const
211  {
212  const auto tokens_to_consume =
213  std::min(tokens.pending_view().size(),
214  positional_arg_t::maximum_count());
215  const auto view = tokens.pending_view().subspan(0, tokens_to_consume);
216 
217  auto result = value_type{};
218  if constexpr (traits::has_push_back_method_v<value_type>) {
219  for (auto token : view) {
220  result.push_back(
221  parent_type::template parse<value_type>(token.name,
222  *this,
223  parents...));
224  }
225  } else if (!view.empty()) {
226  result = parent_type::template parse<value_type>(view.front().name,
227  *this,
228  parents...);
229  }
230 
231  // Pop the tokens, we don't need them anymore
232  tokens.mark_as_processed(view.size());
233 
234  // Validation
235  utility::tuple_type_iterator<policies_type>([&](auto i) {
236  using policy_type = std::tuple_element_t<i, policies_type>;
237  if constexpr (policy::has_validation_phase_method_v<policy_type,
238  value_type>) {
239  this->policy_type::validation_phase(result, *this, parents...);
240  }
241  });
242 
243  // No routing phase, a positional_arg cannot be used as a top-level node
244 
245  return result;
246  }
247
