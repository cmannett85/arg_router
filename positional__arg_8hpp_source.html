137  [[nodiscard]] value_type parse(parsing::parse_target target, const Parents&... parents) const
138  {
139  auto result = value_type{};
140  if constexpr (traits::has_push_back_method_v<value_type>) {
141  for (auto token : target.tokens()) {
142  result.push_back(
143  parent_type::template parse<value_type>(token.name, *this, parents...));
144  }
145  } else if (!target.tokens().empty()) {
146  result = parent_type::template parse<value_type>(target.tokens().front().name,
147  *this,
148  parents...);
149  }

155  this->policy_type::validation_phase(result, *this, parents...);
