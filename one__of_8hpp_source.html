36  constexpr auto num_policies = std::tuple_size_v<policies>;

38  // Find display name
39  constexpr auto display_index =
40  boost::mp11::mp_find_if<policies,

56  S_((std::tuple_element_t<short_index,

60  S_((std::tuple_element_t<long_index,

62  }
63  } else {
