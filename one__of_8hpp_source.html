75  template <typename Validator, bool HasTarget, typename... Parents>
76  [[nodiscard]] std::optional<parsing::parse_target> pre_parse(
77  parsing::pre_parse_data<Validator, HasTarget> pre_parse_data,
78  const Parents&... parents) const
79  {
80  // We need to wrap the pre_parse_data validator so that we can catch child type mismatches
81  // before the caller's validation, otherwise the user can get misleading error messages
82  const auto validator = [&](const auto& child, const auto&... parents) {
83  using child_type = std::decay_t<decltype(child)>;
84 
85  auto type_hash = utility::type_hash<child_type>();
86  if (last_typeid_) {
87  if (type_hash != *last_typeid_) {
88  throw multi_lang_exception{error_code::one_of_selected_type_mismatch,
89  parsing::node_token_type<one_of_t>()};
90  }
91  } else {
92  last_typeid_ = type_hash;
93  }
94 
95  return pre_parse_data.validator()(child, parents...);
96  };
97  auto wrapper =
98  std::optional<parsing::pre_parse_data<std::decay_t<decltype(validator)>, HasTarget>>{};
99  if constexpr (pre_parse_data.has_target) {
100  wrapper =
101  parsing::pre_parse_data{pre_parse_data.args(), pre_parse_data.target(), validator};
102  } else {
103  wrapper = parsing::pre_parse_data{pre_parse_data.args(), validator};
104  }
105 
106  auto found = std::optional<parsing::parse_target>{};
107  utility::tuple_iterator(
108  [&](auto /*i*/, const auto& child) {
109  if (!found) {
110  found = child.pre_parse(*wrapper, parents...);
111  if (found) {
112  // There is no requirement to use pre_parse_data validation, so we also
113  // need to manually check here
114  validator(child, parents...);
115  }
116  }
117  },
118  this->children());
119 
120  return found;
121  }
