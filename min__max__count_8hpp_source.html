22 template <typename MinType, typename MaxType>
23 class min_max_count_t
24 {
25  static_assert(traits::has_value_type_v<MinType> && traits::has_value_type_v<MaxType>,
26  "MinType and MaxType must have a value_type");
27  static_assert(std::is_convertible_v<typename MinType::value_type, std::size_t> &&
28  std::is_convertible_v<typename MaxType::value_type, std::size_t>,
29  "MinType and MaxType must have a value_type that is implicitly "
30  "convertible to std::size_t");
31  static_assert(std::is_integral_v<typename MinType::value_type> &&
32  std::is_integral_v<typename MaxType::value_type>,
33  "MinType and MaxType value_types must be integrals");
34  static_assert((MinType::value >= 0) && (MaxType::value >= 0),
35  "MinType and MaxType must have a value that is a positive number");
36  static_assert(MinType::value <= MaxType::value,
37  "MinType must be less than or equal to MaxType");
38

41  constexpr static auto priority = std::size_t{750};
42

45 
47  [[nodiscard]] constexpr static std::size_t maximum_count() noexcept { return MaxType::value; }

66  template <typename ProcessedTarget, typename... Parents>
67  [[nodiscard]] parsing::pre_parse_result pre_parse_phase(
68  parsing::dynamic_token_adapter& tokens,
69  [[maybe_unused]] utility::compile_time_optional<ProcessedTarget> processed_target,
70  [[maybe_unused]] parsing::parse_target& target,
71  [[maybe_unused]] const Parents&... parents) const
72  {
73  static_assert((sizeof...(Parents) >= 1), "At least one parent needed for min_max_count_t");
74 
75  using owner_type = boost::mp11::mp_first<std::tuple<Parents...>>;
76 
77  // The min/max values are for argument counts, and so need adjusting to accomodate the label
78  // tokens
79  constexpr auto min_count = minimum_count() + (owner_type::is_named ? 1 : 0);
80  constexpr auto max_count = maximum_count() == 0 ?
81  maximum_count() :
82  (maximum_count() - (owner_type::is_named ? 0 : 1));
83 
84  // If we are unnamed then check that tokens haven't already been parsed for the owner. This
85  // is for the scenario where there are multiple positional arg-like types under a mode, and
86  // once one has been processed, then it should be skipped so subsequent ones are used
87  // instead
88  if constexpr (!owner_type::is_named) {
89  static_assert(!processed_target.empty, "processed_target cannot be empty");
90 
91  if (has_filled_tokens(*processed_target,
92  std::type_index{typeid(owner_type)},
93  max_count)) {
94  return parsing::pre_parse_action::skip_node;
95  }
96  }
97 
98  // Check that we are in the bounds. We can't check that we have exceeded the maximum
99  // because there may be other nodes that will consume tokens, so this node will take its
100  // maximum (potentially) and any left over tokens will be trigger an error later in the
101  // processing
102  if (tokens.size() < min_count) {
103  return parse_exception{"Minimum count not reached",
104  parsing::node_token_type<owner_type>()};
105  }
106 
107  // Transfer any remaining up to the maximum count
108  tokens.transfer(tokens.begin() + std::min(max_count, tokens.size()));
109 
110  return parsing::pre_parse_action::valid_node;
111  }
112 
113 private:
114  [[nodiscard]] static bool has_filled_tokens(const parsing::parse_target& target,
115  std::type_index owner_index,
116  std::size_t max_count) noexcept
117  {
118  if (target.node_type() == owner_index) {
119  return target.tokens().size() >= max_count;
120  }
121 
122  for (const auto& sub_target : target.sub_targets()) {
123  if (has_filled_tokens(sub_target, owner_index, max_count)) {
124  return true;
125  }
126  }
127 
128  return false;
129  }
130 };
