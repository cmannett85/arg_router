22 template <typename MinType, typename MaxType>
23 class min_max_count_t
24 {
25  static_assert(traits::has_value_type_v<MinType> &&
26  traits::has_value_type_v<MaxType>,
27  "MinType and MaxType must have a value_type");
28  static_assert(
29  std::is_convertible_v<typename MinType::value_type, std::size_t> &&
30  std::is_convertible_v<typename MaxType::value_type, std::size_t>,
31  "MinType and MaxType must have a value_type that is implicitly "
32  "convertible to std::size_t");
33  static_assert(std::is_integral_v<typename MinType::value_type> &&
34  std::is_integral_v<typename MaxType::value_type>,
35  "MinType and MaxType value_types must be integrals");
36  static_assert(
37  (MinType::value >= 0) && (MaxType::value >= 0),
38  "MinType and MaxType must have a value that is a positive number");
39  static_assert(MinType::value <= MaxType::value,
40  "MinType must be less than or equal to MaxType");
41 
42 public:
44  constexpr static auto priority = std::size_t{750};
45 
47  [[nodiscard]] constexpr static std::size_t minimum_count() noexcept
48  {
49  return MinType::value;
50  }
51 
53  [[nodiscard]] constexpr static std::size_t maximum_count() noexcept
54  {
55  return MaxType::value;
56  }
57 
75  template <typename... Parents>
76  [[nodiscard]] parsing::pre_parse_result pre_parse_phase(
77  parsing::dynamic_token_adapter& tokens,
78  [[maybe_unused]] parsing::token_list::pending_view_type
79  processed_tokens,
80  [[maybe_unused]] const Parents&... parents) const
81  {
82  static_assert((sizeof...(Parents) >= 1),
83  "At least one parent needed for min_max_count_t");
84 
85  using owner_type = boost::mp11::mp_first<std::tuple<Parents...>>;
86 
87  constexpr auto min_count =
88  minimum_count() + (owner_type::is_named ? 1 : 0);
89  constexpr auto max_count = maximum_count();
90 
91  // Check that we are in the bounds. We can't check that we have
92  // exceeded the maximum because there may be other nodes that will
93  // consume tokens, so this node will take its maximum (potentially) and
94  // any left over tokens will be trigger an error later in the processing
95  if (tokens.size() < min_count) {
96  return parse_exception{"Minimum count not reached",
97  parsing::node_token_type<owner_type>()};
98  }
99 
100  // Transfer any remaining up to the maximum count
101  tokens.transfer(tokens.begin() + std::min(max_count, tokens.size()));
102 
103  return parsing::pre_parse_action::valid_node;
104  }
105 };
