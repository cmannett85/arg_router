24 template <typename MinType, typename MaxType>
25 class min_max_count_t
26 {
27  static_assert(traits::has_value_type_v<MinType> && traits::has_value_type_v<MaxType>,
28  "MinType and MaxType must have a value_type");
29  static_assert(std::is_convertible_v<typename MinType::value_type, std::size_t> &&
30  std::is_convertible_v<typename MaxType::value_type, std::size_t>,
31  "MinType and MaxType must have a value_type that is implicitly "
32  "convertible to std::size_t");
33  static_assert(std::is_integral_v<typename MinType::value_type> &&
34  std::is_integral_v<typename MaxType::value_type>,
35  "MinType and MaxType value_types must be integrals");
36  static_assert((MinType::value >= 0) && (MaxType::value >= 0),
37  "MinType and MaxType must have a value that is a positive number");
38  static_assert(MinType::value <= MaxType::value,
39  "MinType must be less than or equal to MaxType");
40 
41 public:
43  constexpr static auto priority = std::size_t{750};
44 
46  [[nodiscard]] constexpr static std::size_t minimum_count() noexcept { return MinType::value; }
47 
49  [[nodiscard]] constexpr static std::size_t maximum_count() noexcept { return MaxType::value; }

69  template <typename ProcessedTarget, typename... Parents>
70  [[nodiscard]] parsing::pre_parse_result pre_parse_phase(
71  parsing::dynamic_token_adapter& tokens,
72  [[maybe_unused]] utility::compile_time_optional<ProcessedTarget> processed_target,
73  [[maybe_unused]] parsing::parse_target& target,
74  [[maybe_unused]] const Parents&... parents) const
75  {
76  static_assert((sizeof...(Parents) >= 1), "At least one parent needed for min_max_count_t");
77 
78  using owner_type = boost::mp11::mp_first<std::tuple<Parents...>>;
79 
80  // The min/max values are for argument counts, and so need adjusting to accomodate the label
81  // tokens
82  constexpr auto mn_count = minimum_count() + (owner_type::is_named ? 1 : 0);
83  constexpr auto mx_count = maximum_count() == 0 ?
84  maximum_count() :
85  (maximum_count() - (owner_type::is_named ? 0 : 1));
86 
87  // If we are unnamed then check that tokens haven't already been parsed for the owner. This
88  // is for the scenario where there are multiple positional arg-like types under a mode, and
89  // once one has been processed, then it should be skipped so subsequent ones are used
90  // instead
91  if constexpr (!owner_type::is_named) {
92  static_assert(!processed_target.empty, "processed_target cannot be empty");
93 
94  if (has_filled_tokens(*processed_target, utility::type_hash<owner_type>(), mx_count)) {
95  return parsing::pre_parse_action::skip_node;
96  }
97  }
98 
99  // Check that we are in the bounds. We can't check that we have exceeded the maximum
100  // because there may be other nodes that will consume tokens, so this node will take its
101  // maximum (potentially) and any left over tokens will be trigger an error later in the
102  // processing
103  if (tokens.size() < mn_count) {
104  return multi_lang_exception{error_code::minimum_count_not_reached,
105  parsing::node_token_type<owner_type>()};
106  }
107 
108  // Transfer any remaining up to the maximum count
109  tokens.transfer(tokens.begin() + std::min(mx_count, tokens.size()));
110 
111  return parsing::pre_parse_action::valid_node;
112  }
113 
114 private:
115  [[nodiscard]] static bool has_filled_tokens(const parsing::parse_target& target,
116  std::size_t owner_hash_code,
117  std::size_t mx_count) noexcept
118  {
119  if (target.node_type() == owner_hash_code) {
120  return target.tokens().size() >= mx_count;
121  }
122 
123  for (const auto& sub_target : target.sub_targets()) {
124  if (has_filled_tokens(sub_target, owner_hash_code, mx_count)) {
125  return true;
126  }
127  }
128 
129  return false;
130  }
131 };
