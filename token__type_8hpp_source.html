239  template <typename InputIt>
240  pending_view_type::const_iterator //
241  insert_pending(pending_view_type::const_iterator pos,
242  InputIt first,
243  InputIt last)
244  {
245  // If there's enough space, try to reclaim the leading memory first
246  const auto input_count =
247  static_cast<std::size_t>(std::distance(first, last));
248 
249  if ((pos == pending_view().begin()) && (input_count <= head_offset_)) {
250  const auto result = pending_view().begin();
251  head_offset_ -= input_count;
252  std::move(first, last, data_.begin() + head_offset_);
253  return result;
254  } else {
255  // We have to do this nonsense because the vector may reallocate
256  // due to the insertion which would invalidate pos+input_count
257  const auto offset = std::distance(pending_view().begin(), pos);
258  data_.insert(data_.begin() + head_offset_ + offset, first, last);
259  return pending_view().begin() + offset + input_count;
260  }
261  }
