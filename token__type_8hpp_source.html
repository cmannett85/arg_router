239  insert_pending(pending_view_type::const_iterator pos,
240  InputIt first,
241  InputIt last)
242  {
243  // If there's enough space, try to reclaim the leading memory first
244  const auto input_count =
245  static_cast<std::size_t>(std::distance(first, last));
246 
247  if ((pos == pending_view().begin()) && (input_count <= head_offset_)) {
248  const auto result = pending_view().begin();
249  head_offset_ -= input_count;
250  std::move(first, last, data_.begin() + head_offset_);
251  return result;
252  } else {
253  // We have to do this nonsense because the vector may reallocate
254  // due to the insertion which would invalidate pos+input_count
255  const auto offset = std::distance(pending_view().begin(), pos);
256  data_.insert(data_.begin() + head_offset_ + offset, first, last);
257  return pending_view().begin() + offset + input_count;
