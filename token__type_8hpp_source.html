246  template <typename InputIt>
247  pending_view_type::const_iterator //
248  insert_pending(pending_view_type::const_iterator pos,
249  InputIt first,
250  InputIt last)
251  {
252  // If there's enough space, try to reclaim the leading memory first
253  const auto input_count =
254  static_cast<std::size_t>(std::distance(first, last));
255 
256  if ((pos == pending_view().begin()) && (input_count <= head_offset_)) {
257  const auto result = pending_view().begin();
258  head_offset_ -= input_count;
259  std::move(first, last, data_.begin() + head_offset_);
260  return result;
261  } else {
262  // We have to do this nonsense because the vector may reallocate
263  // due to the insertion which would invalidate pos+input_count
264  const auto offset = std::distance(pending_view().begin(), pos);
265  data_.insert(data_.begin() + head_offset_ + offset, first, last);
266  return pending_view().begin() + offset + input_count;
267  }
268  }
