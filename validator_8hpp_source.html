175  template <typename T, typename... Parents>
176  constexpr static void check()
177  {
178  using ParentTuple = std::tuple<Parents...>;
179  constexpr auto NumParents = sizeof...(Parents);
180 
181  // Make sure there's at least one parent beyond the owner
182  if constexpr (NumParents > 1) {
183  // Find a mode type, if there's one present we stop moving up through
184  // the ancestors at that point, otherwise we go up to the root
185  constexpr auto mode_index =
186  algorithm::find_specialisation_v<ModeType, ParentTuple>;
187  using start_type = boost::mp11::mp_eval_if_c<
188  mode_index == NumParents,
189  boost::mp11::mp_back<ParentTuple>,
190  boost::mp11::mp_at,
191  ParentTuple,
192  traits::integral_constant<mode_index>>;
193 
194  using path_type =
195  boost::mp11::mp_take_c<ParentTuple,
196  std::min(mode_index + 1, NumParents)>;
197 
198  // Recurse the tree from the oldest generation, testing that no
199  // other policy matches ours
200  utility::tree_recursor<checker<T, path_type>, start_type>();
201  }
