177  template <typename T, typename... Parents>
178  constexpr static void check()
179  {
180  using ParentTuple = std::tuple<Parents...>;
181  constexpr auto NumParents = sizeof...(Parents);
182 
183  // Make sure there's at least one parent beyond the owner
184  if constexpr (NumParents > 1) {
185  // Find a mode type, if there's one present we stop moving up through
186  // the ancestors at that point, otherwise we go up to the root
187  constexpr auto mode_index =
188  algorithm::find_specialisation_v<ModeType, ParentTuple>;
189  using start_type = boost::mp11::mp_eval_if_c<
190  mode_index == NumParents,
191  boost::mp11::mp_back<ParentTuple>,
192  boost::mp11::mp_at,
193  ParentTuple,
194  traits::integral_constant<mode_index>>;
195 
196  using path_type =
197  boost::mp11::mp_take_c<ParentTuple,
198  std::min(mode_index + 1, NumParents)>;
199 
200  // Recurse the tree from the oldest generation, testing that no
201  // other policy matches ours
202  utility::tree_recursor<checker<T, path_type>, start_type>();
203  }
204  }
