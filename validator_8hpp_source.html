163 struct policy_unique_from_owner_parent_to_mode_or_root {
164  template <typename Policy, typename PathToThis>
165  struct checker {
166  template <typename Current, typename... Parents>
167  constexpr static void fn()
168  {
169  using path_type = std::tuple<Parents...>;
170 
171  // Skip checking ourself
172  if constexpr (!std::is_same_v<PathToThis, path_type>) {
173  static_assert(!std::is_same_v<Policy, Current>,
174  "Policy must be unique in the parse tree up to "
175  "the nearest mode or root");
176  }
177  }

179 
180  template <typename T, typename... Parents>
181  constexpr static void check()
182  {
183  using ParentTuple = std::tuple<Parents...>;
184  constexpr auto NumParents = sizeof...(Parents);
185 
186  // Make sure there's at least one parent beyond the owner
187  if constexpr (NumParents > 1) {
188  // Find a mode type, if there's one present we stop moving up through
189  // the ancestors at that point, otherwise we go up to the root
190  constexpr auto mode_index =
191  algorithm::find_specialisation_v<ModeType, ParentTuple>;
192  using start_type = boost::mp11::mp_eval_if_c<
193  mode_index == NumParents,
194  boost::mp11::mp_back<ParentTuple>,
195  boost::mp11::mp_at,
196  ParentTuple,
197  traits::integral_constant<mode_index>>;
198 
199  using path_type =
200  boost::mp11::mp_take_c<ParentTuple,
201  std::min(mode_index + 1, NumParents)>;
202 
203  // Recurse the tree from the oldest generation, testing that no
204  // other policy matches ours
205  utility::tree_recursor<checker<T, path_type>, start_type>();
206  }
207  }
208 };
