176  template <typename T, typename... Parents>
177  constexpr static void check()
178  {
179  using ParentTuple = std::tuple<Parents...>;
180  constexpr auto NumParents = sizeof...(Parents);
181 
182  // Make sure there's at least one parent beyond the owner
183  if constexpr (NumParents > 1) {
184  // Find a mode type, if there's one present we stop moving up through
185  // the ancestors at that point, otherwise we go up to the root
186  constexpr auto mode_index =
187  algorithm::find_specialisation_v<ModeType, ParentTuple>;
188  using start_type = boost::mp11::mp_eval_if_c<
189  mode_index == NumParents,
190  boost::mp11::mp_back<ParentTuple>,
191  boost::mp11::mp_at,
192  ParentTuple,
193  traits::integral_constant<mode_index>>;
194 
195  using path_type =
196  boost::mp11::mp_take_c<ParentTuple,
197  std::min(mode_index + 1, NumParents)>;
198 
199  // Recurse the tree from the oldest generation, testing that no
200  // other policy matches ours
201  utility::tree_recursor<checker<T, path_type>, start_type>();
202  }
