119  template <typename Current, typename... Parents>
120  constexpr static void fn() noexcept
121  {
122  // Find the matching rule
123  constexpr auto rule_index = boost::mp11::mp_find_if_q< //
124  rules_type,
125  rule_lookup<Current>>::value;
126  static_assert(rule_index != std::tuple_size_v<rules_type>, "No rule for Current");
127

129  using conditions =
130  boost::mp11::mp_drop_c<std::tuple_element_t<rule_index, rules_type>, 1>;
131  utility::tuple_type_iterator<conditions>([](auto i) {
132  using condition = std::tuple_element_t<i, conditions>;

134  });

219  // Don't recurse into child modes, they effectively have their own namespace
220  struct skipper {
221  template <typename Current, typename...>
222  [[nodiscard]] constexpr static bool fn() noexcept
223  {
224  return is_mode<Current>::value;
225  }
226  };
227 
228  template <typename T, typename... Parents>
229  constexpr static void check() noexcept
230  {
231  using ParentTuple = std::tuple<Parents...>;
232  constexpr auto NumParents = sizeof...(Parents);
233 
234  // Make sure there's at least one parent beyond the owner
235  if constexpr (NumParents > 1) {
236  // Find a mode type, if there's one present we stop moving up through the ancestors at
237  // that point, otherwise we go up to the root
238  constexpr auto mode_index = boost::mp11::mp_find_if<ParentTuple, is_mode>::value;
239  using start_type = boost::mp11::mp_eval_if_c<mode_index == NumParents,
240  boost::mp11::mp_back<ParentTuple>,
241  boost::mp11::mp_at,
242  ParentTuple,
243  traits::integral_constant<mode_index>>;
244 
245  using path_type =
246  boost::mp11::mp_take_c<ParentTuple, std::min(mode_index + 1, NumParents)>;
247 
248  // Recurse the tree from the oldest generation, testing that no other policy matches
249  // ours
250  utility::tree_type_recursor<checker<T, path_type>, skipper, start_type>();
251  }
252  }
253 };

461 template <template <typename...> typename... NodeTypes>
462 struct node_types_must_be_at_end {
463  static_assert(sizeof...(NodeTypes), "Must be at least one node type");
464 
465  template <typename T>
466  using is_target_node =
467  boost::mp11::mp_any_of<std::tuple<traits::is_specialisation_of<T, NodeTypes>...>,
468  boost::mp11::mp_to_bool>;
469 
470  template <typename T, typename...>
471  constexpr static void check() noexcept
472  {
473  // Find the first index of a target node, then for each element type after, check that it
474  // is also the same type of node
475 
476  // Remap the child types to a tuple of booleans, where true means that they are
477  // specialisations of one of the given node types
478  using children_type = typename T::children_type;
479  using is_target_node_map = boost::mp11::mp_transform_q<
480  boost::mp11::mp_bind<boost::mp11::mp_to_bool,
481  boost::mp11::mp_bind<is_target_node, boost::mp11::_1>>,
482 
483  children_type>;
484 
485  constexpr auto first_node_index =
486  boost::mp11::mp_find<is_target_node_map, boost::mp11::mp_true>::value;
487  if constexpr (first_node_index != std::tuple_size_v<is_target_node_map>) {
488  using drop_before_first = boost::mp11::mp_drop_c<is_target_node_map, first_node_index>;
489  static_assert(boost::mp11::mp_all_of<drop_before_first, boost::mp11::mp_to_bool>::value,
490  "Node types must all appear at the end of child list for a node");
491  }
492  }
