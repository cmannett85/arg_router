116  void parse(int argc, char* argv[]) const
117  {
118  // Skip the program name
119  auto args = vector<parsing::token_type>{};
120  args.reserve(argc - 1);
121  for (auto i = 1; i < argc; ++i) {
122  args.emplace_back(parsing::prefix_type::none, argv[i]);
123  }
124 
125  // Take a copy of the front token for the error messages
126  const auto front_token =
127  args.empty() ? //
128  parsing::token_type{parsing::prefix_type::none, ""} :
129  args.front();
130 
131  // Find a matching child
132  auto match = std::optional<parsing::parse_target>{};
133  utility::tuple_iterator(
134  [&](auto /*i*/, const auto& child) {
135  // Skip any remaining children if one has been found
136  if (!match &&
137  (match = child.pre_parse(parsing::pre_parse_data{args},
138  *this))) {
139  if (!args.empty()) {
140  throw parse_exception{"Unhandled arguments", args};
141  }
142  (*match)();
143  }
144  },
145  this->children());
146  if (!match) {
147  if (front_token.name.empty()) {
148  throw parse_exception{"No arguments passed"};
149  } else {
150  throw parse_exception{"Unknown argument", front_token};
151  }
152  }
