117  void parse(int argc, char* argv[]) const

124 
125  // Find the initial child, if there are no tokens then create an empty
126  // one - an anonymous mode will still be found
127  const auto first_token =
128  tokens.pending_view().empty() ?
129  parsing::token_type{parsing::prefix_type::NONE, ""} :
130  tokens.pending_view().front();
131  const auto found_child =
132  parent_type::find(first_token,
133  [&](auto /*i*/, const auto& child) { //
134  child.parse(tokens, *this);
135  });
136  if (!found_child) {

138  throw parse_exception{"No arguments passed"};
