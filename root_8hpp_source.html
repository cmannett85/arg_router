119  static void top_level_visitor(const Node& node, parsing::token_list tokens)
120  {
121  using node_type = std::decay_t<Node>;
122 
123  // From this point on, we don't care about the non-mode children, so
124  // filter them out
125  const auto non_mode_children =
126  algorithm::tuple_filter_and_construct< //
127  boost::mp11::mp_bind<boost::mp11::mp_not,
128  boost::mp11::mp_bind< //
129  node_category::is_generic_mode_like,
130  boost::mp11::_1>>::template fn>(
131  node.children());
132 
133  // Build results tuple
134  auto router_args = parsing::optional_router_args_t<node_type>{};
135 
136  // Mode-like types need handling separately
137  constexpr auto non_mode_children_size =
138  std::tuple_size_v<std::decay_t<decltype(non_mode_children)>>;
139  if constexpr (non_mode_children_size > 0) {
140  constexpr auto router_args_size =
141  std::tuple_size_v<std::decay_t<decltype(router_args)>>;
142  static_assert(non_mode_children_size == router_args_size,
143  "Non-mode children count must match router arg size");
144 
145  // Process the tokens
146  while (!tokens.empty()) {
147  const auto found_child = parsing::visit_child(
148  tokens.front(),
149  non_mode_children,
150  router_args,
151  [&](auto i, const auto& child) {
152  process_token<i, Node>(child, router_args, tokens);
153  });
154 
155  if (!found_child) {
156  throw parse_exception{"Unknown argument", tokens.front()};
157  }
158  }
159 
160  auto fra = filter_aliased_results<Node>(router_args);
161  check_hits(node, fra);
162  call_router(node, fra, tokens);
163  } else {
164  process_token<0, Node>(node, router_args, tokens);
165  call_router(node, router_args, tokens);
166  }

169  template <std::size_t I,
170  typename Parent,
171  typename Node,
172  typename RouterArgs>
173  static void process_token(const Node& node,
174  RouterArgs& router_args,
175  parsing::token_list& tokens)
176  {
177  auto& arg = std::get<I>(router_args);
178 
179  if constexpr (policy::has_value_tokens_v<Node>) {
180  // Drop the token if present, the value follows
181  if constexpr (traits::has_match_method_v<Node>) {
182  tokens.pop_front();
183  }
184  auto value = parse_token_argument(node, tokens.front().name);
185 
186  if constexpr (traits::has_aliased_policies_type_v<Node>) {
187  process_aliased_token<Node, Parent>(value, router_args);
188  } else if constexpr (supports_multiple_values<Node>()) {
189  if (!arg) {
190  arg = {std::move(value)};
191  } else {
192  arg->push_back(std::move(value));
193  }
194 
195  // Positional args have contiguous value tokens
196  if constexpr (policy::has_contiguous_value_tokens_v<Node>) {
197  const auto count = contiguous_token_count<Node>(tokens);
198  for (auto i = 1u; i < count; ++i) {
199  tokens.pop_front();
200  value = parse_token_argument(node, tokens.front().name);
201  arg->push_back(std::move(value));
202  }
203  }
204  } else {
205  if (arg) {
206  throw parse_exception{"Argument has already been set",
207  tokens.front()};
208  }
209 
210  arg = std::move(value);
211  }
212  } else if constexpr (node_category::is_flag_like_v<Node>) {
213  if constexpr (traits::has_aliased_policies_type_v<Node>) {
214  process_aliased_token<Node, Parent>(true, router_args);
215  } else {
216  if (arg) {
217  throw parse_exception{"Argument has already been set",
218  tokens.front()};
219  }
220 
221  // We have hit the flag, so set the value to true
222  arg = true;
223  }
224  } else if constexpr (node_category::is_counting_flag_like_v<Node>) {
225  // Counting flag
226  } else {
227  static_assert(traits::always_false_v<Node>,
228  "Unhandled node category");
229  }
230 
231  tokens.pop_front();
232  }
