146  void parse(int argc, char** argv) const
147  {
148  try {
149  // Skip the program name
150  auto args = vector<parsing::token_type>{};
151  args.reserve(argc - 1);
152  for (auto i = 1; i < argc; ++i) {
153  args.emplace_back(parsing::prefix_type::none, argv[i]);
154  }
155 
156  // Take a copy of the front token for the error messages
157  const auto front_token = args.empty() ? //
158  parsing::token_type{parsing::prefix_type::none, ""} :
159  args.front();
160 
161  // Find a matching child
162  auto match = std::optional<parsing::parse_target>{};
163  utility::tuple_iterator(
164  [&](auto /*i*/, const auto& child) {
165  // Skip any remaining children if one has been found
166  if (!match && (match = child.pre_parse(parsing::pre_parse_data{args}, *this))) {
167  if (!args.empty()) {
168  throw multi_lang_exception{error_code::unhandled_arguments, args};
169  }
170  (*match)();
171  }
172  },
173  this->children());
174  if (!match) {
175  if (front_token.name.empty()) {
176  throw multi_lang_exception{error_code::no_arguments_passed};
177  }
178  throw multi_lang_exception{error_code::unknown_argument, front_token};
179  }
180  } catch (multi_lang_exception& e) {
181  // Convert the error code exception to a parse_exception. This method will always be
182  // present because even if an exception_translator-like policy is not specified by the
183  // user, a default en_GB one is added
184  this->translate_exception(e);
185  }
186  }

193  void help([[maybe_unused]] std::ostream& stream) const
194  {
195  // Have to lambda wrap here due to an MSVC bug where the false if constexpr branch is
196  // evaluated
197  [&](auto* type_ptr) {
198  using root_type = std::remove_pointer_t<decltype(type_ptr)>;
199 
200  constexpr static auto help_index =
201  boost::mp11::mp_find_if<typename root_type::children_type,
202  traits::has_generate_help_method>::value;
203 
204  if constexpr (help_index < std::tuple_size_v<typename root_type::children_type>) {
205  using help_type =
206  std::tuple_element_t<help_index, typename root_type::children_type>;
207  constexpr auto flatten =
208  algorithm::has_specialisation_v<policy::flatten_help_t,
209  typename help_type::policies_type>;
210 
211  try {
212  std::get<help_index>(this->children())
213  .template generate_help<root_type, help_type, flatten>(stream);
214  } catch (multi_lang_exception& e) {
215  this->translate_exception(e);
216  }
217  }
218  }(static_cast<root_t*>(nullptr));
219  }
