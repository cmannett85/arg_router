104  void parse(int argc, char** argv) const
105  {
106  // Skip the program name
107  auto args = vector<parsing::token_type>{};
108  args.reserve(argc - 1);
109  for (auto i = 1; i < argc; ++i) {
110  args.emplace_back(parsing::prefix_type::none, argv[i]);
111  }
112 
113  // Take a copy of the front token for the error messages
114  const auto front_token = args.empty() ? //
115  parsing::token_type{parsing::prefix_type::none, ""} :
116  args.front();
117 
118  // Find a matching child
119  auto match = std::optional<parsing::parse_target>{};
120  utility::tuple_iterator(
121  [&](auto /*i*/, const auto& child) {
122  // Skip any remaining children if one has been found
123  if (!match && (match = child.pre_parse(parsing::pre_parse_data{args}, *this))) {
124  if (!args.empty()) {
125  throw parse_exception{"Unhandled arguments", args};
126  }
127  (*match)();
128  }
129  },
130  this->children());
131  if (!match) {
132  if (front_token.name.empty()) {
133  throw parse_exception{"No arguments passed"};
134  }
135  throw parse_exception{"Unknown argument", front_token};
136  }
137  }
