97  const auto separator_index = first_token.name.find_first_of(value_separator());
98  if (separator_index == std::string_view::npos) {
99  return parsing::pre_parse_action::skip_node;

103  if (value_arg.empty()) {
104  return parsing::pre_parse_action::skip_node;
105  }
106

108  tokens.insert(first + 1, {parsing::prefix_type::none, value_arg});
109 
110  // Remove the value and separator part of the label token
111  const auto label_arg = first_token.name.substr(0, separator_index);
112  first.set({first_token.prefix, label_arg});
