56  [[nodiscard]] parsing::pre_parse_result pre_parse_phase(
57  parsing::dynamic_token_adapter& tokens,
58  [[maybe_unused]] utility::compile_time_optional<ProcessedTarget> processed_target,

61  {
62  static_assert((sizeof...(Parents) >= 1),
63  "At least one parent needed for value_separator_t");
64 
65  using owner_type = boost::mp11::mp_first<std::tuple<Parents...>>;
66 
67  static_assert(traits::has_minimum_count_method_v<owner_type> &&
68  traits::has_maximum_count_method_v<owner_type>,
69  "Value separator support requires an owning node to have "
70  "minimum and maximum count policies");
71  static_assert((owner_type::minimum_count() == 1) && (owner_type::maximum_count() == 1),
72  "Value separator support requires an owning node to have "
73  "a fixed count of 1");
74 
75  auto first = tokens.begin();
76  const auto first_token = *first;
77 
78  // Find the separator
79  const auto separator_index = first_token.name.find_first_of(value_separator());
80  if (separator_index == std::string_view::npos) {
81  return parsing::pre_parse_action::skip_node;
82  }
83 
84  const auto value_arg = first_token.name.substr(separator_index + S::get().size());
85  if (value_arg.empty()) {
86  return parsing::pre_parse_action::skip_node;
87  }
88 
89  // Insert value token after the label one
90  tokens.insert(first + 1, {parsing::prefix_type::none, value_arg});
91 
92  // Remove the value and separator part of the label token
93  const auto label_arg = first_token.name.substr(0, separator_index);
94  first.set({first_token.prefix, label_arg});
95 
96  return parsing::pre_parse_action::valid_node;
