171 [[nodiscard]] inline constexpr std::optional<type> decode(std::string_view str) noexcept
172 {
173  const auto bytes_to_read = size(str);
174  if (bytes_to_read == 0) {
175  return {};
176  }
177 
178  if (bytes_to_read == 1) {
179  // ASCII
180  return str[0];
181  }
182 
183  if (str.size() < bytes_to_read) {
184  return {};
185  }
186 
187  constexpr auto subsequent_byte_data_bits = 6;
188  constexpr auto subsequent_byte_mask = type{(1 << subsequent_byte_data_bits) - 1};
189  constexpr auto maximum_first_byte_data_mask = type{0b0001'1111};
190 
191  const auto first_byte = static_cast<std::uint8_t>(str[0]);
192 
193  auto result = first_byte & (maximum_first_byte_data_mask >> (bytes_to_read - 2));
194  for (auto i = 1u; i < bytes_to_read; ++i) {
195  const auto subsequent_byte = static_cast<type>(str[i]);
196 
197  // Move the previous reads up to make space for the subsequent byte's data
198  result <<= subsequent_byte_data_bits;
199  result |= subsequent_byte & subsequent_byte_mask;
200  }
201 
202  return result;
