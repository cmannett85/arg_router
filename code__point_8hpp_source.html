107 [[nodiscard]] inline constexpr std::optional<type> decode(std::string_view str) noexcept
108 {
109  const auto bytes_to_read = size(str);
110  if (bytes_to_read == 0) {
111  return {};
112  }
113 
114  if (bytes_to_read == 1) {
115  // ASCII
116  return str[0];
117  }
118 
119  if (str.size() < bytes_to_read) {
120  return {};
121  }
122 
123  constexpr auto subsequent_byte_data_bits = 6;
124  constexpr auto subsequent_byte_mask = type{(1 << subsequent_byte_data_bits) - 1};
125  constexpr auto maximum_first_byte_data_mask = type{0b0001'1111};
126 
127  const auto first_byte = static_cast<std::uint8_t>(str[0]);
128 
129  auto result = first_byte & (maximum_first_byte_data_mask >> (bytes_to_read - 2));
130  for (auto i = 1u; i < bytes_to_read; ++i) {
131  const auto subsequent_byte = static_cast<type>(str[i]);
132 
133  // Move the previous reads up to make space for the subsequent byte's data
134  result <<= subsequent_byte_data_bits;
135  result |= subsequent_byte & subsequent_byte_mask;
136  }
137 
138  return result;
139 }
