457  void transfer(iterator it)
458  {
459  // If the iterator is an end(), then consume all the unprocessed tokens
460  if (it.is_end()) {
461  it.i_ = size() - 1;
462  }
463 
464  if ((it.i_ < 0) || (it.i_ < static_cast<iterator::difference_type>(
465  processed_->size()))) {
466  return;

470  processed_->insert(processed_->end(),
471  unprocessed_->begin(),
