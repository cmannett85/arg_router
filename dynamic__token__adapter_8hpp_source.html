416  void transfer(iterator it)
417  {
418  // If the iterator is an end(), then consume all the unprocessed tokens
419  if (it.is_end()) {
420  it.i_ = size() - 1;
421  }
422 
423  if ((it.i_ < 0) || (it.i_ < static_cast<iterator::difference_type>(processed_->size()))) {
424  return;
425  }
426

428  processed_->insert(processed_->end(), unprocessed_->begin(), unprocessed_->begin() + count);
429  unprocessed_->erase(unprocessed_->begin(), unprocessed_->begin() + count);
