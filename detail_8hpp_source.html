17 // Collect the child names and concetenate into a compile time string for the one_of display name
18 template <typename ParentDocName, typename... Params>
19 struct generate_string_of_child_names {
20  template <typename Prefix, typename String>
21  using prepend_prefix =
22  boost::mp11::mp_flatten<boost::mp11::mp_insert_c<String, 0, typename Prefix::array_type>>;

24  template <typename Child>
25  struct build_name {
26  // This is clunky, but it's still more readable than doing it with bare types
27  constexpr static auto build()
28  {
29  using policies = typename Child::policies_type;
30  constexpr auto num_policies = std::tuple_size_v<policies>;
31 
32  // Find display name
33  constexpr auto display_index =
34  boost::mp11::mp_find_if<policies, traits::has_display_name_method>::value;
35  if constexpr (display_index >= num_policies) {
36  // Find long name
37  constexpr auto long_index =
38  boost::mp11::mp_find_if<policies, traits::has_long_name_method>::value;
39  if constexpr (long_index >= num_policies) {
40  // Find short name
41  constexpr auto short_index =
42  boost::mp11::mp_find_if<policies, traits::has_short_name_method>::value;
43 
44  static_assert(short_index < num_policies, "All children must be named");
45 
46  return S_(config::short_prefix){} +
47  S_((std::tuple_element_t<short_index, policies>::short_name())){};
48  } else {
49  return S_(config::long_prefix){} +
50  S_((std::tuple_element_t<long_index, policies>::long_name())){};
51  }
52  } else {
53  return S_((std::tuple_element_t<display_index, policies>::display_name())){};
54  }
55  }
56 
57  using type = typename std::decay_t<decltype(build())>::array_type;
58  };
59 
60  using children_type = boost::mp11::mp_filter<is_tree_node, std::tuple<Params...>>;
61 
62  // The gist is that for each child we get its name (prepended with an appropriate prefix) and
63  // concatenate with a comma separator. And then add a helpful prefix to the whole thing
64  using string_array = prepend_prefix<
65  ParentDocName,
66  boost::mp11::mp_join<
67  boost::mp11::mp_transform_q<
68  boost::mp11::mp_bind<traits::get_type,
69  boost::mp11::mp_bind<build_name, boost::mp11::_1>>,
70  children_type>,
71  traits::integral_constant<','>>>;
72 
73  using type = utility::convert_to_cts_t<string_array>;
74 };
